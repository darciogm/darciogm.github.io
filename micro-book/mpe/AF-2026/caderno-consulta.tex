\bloco{Bloco 1 --- Teoria do Consumidor (Aulas 1--3)}

\sub{Axiomas e representação}
\emph{Completude}, \emph{transitividade} (juntas: racionalidade), \emph{monotonicidade}, \emph{convexidade} (gosto por misturas), \emph{continuidade} ($\{y:y\succeq x\}$ e $\{y:x\succeq y\}$ fechados). Debreu: $\succeq$ completa, transitiva e \emph{contínua} $\Rightarrow$ existe $u$ contínua que a representa (hipóteses \emph{suficientes}). Utilidade é \emph{ordinal}: transformação estritamente crescente preserva $\succeq$.

\sub{UMP, formas funcionais e tangência}
$\max u$ s.a.\ $p\cdot x=m$: interior dá tangência $\text{TMS}_{12}=p_1/p_2$ \emph{mais} a restrição $\rightarrow$ marshalliana $x^M(p,m)$ e indireta $v(p,m)=u(x^M)$. \emph{Cobb-Douglas} $u=x_1^{\alpha}x_2^{1-\alpha}$: $x_1^M=\dfrac{\alpha m}{p_1}$ (participações $s_i$ fixas no gasto). \emph{Quase-linear} $u=\varphi(x_1)+x_2$ (interior): $\varphi'(x_1)=p_1/p_2$, sem $m$ --- com $\varphi=\theta\ln x_1$, a CPO $\theta/x_1=p_1/p_2$. \emph{CES} $\big(\alpha x_1^{\rho}+(1-\alpha)x_2^{\rho}\big)^{1/\rho}$ é homotética; $\sigma=1/(1-\rho)$. Para passar de $v$ a $e$: resolva $v(p,m)=\bar u$ em $m$ (inversão).

\sub{Dualidade --- mapa completo}
EMP: $\min p\cdot x$ s.a.\ $u(x)\ge\bar u$ $\rightarrow$ hicksiana $h(p,\bar u)$, dispêndio $e(p,\bar u)=p\cdot h$.
\[
e(p,v(p,m))=m,\quad x^M(p,m)=h(p,v(p,m)).
\]
\emph{Roy}: $x_i^M=-\dfrac{\partial v/\partial p_i}{\partial v/\partial m}$.\quad \emph{Shephard}: $h_i=\dfrac{\partial e}{\partial p_i}$. $e$ côncava e homog.\ grau 1 em $p$; $v,x^M$ homog.\ grau 0 em $(p,m)$.
\guia{Diante de uma $v(p,m)$ dada, monte $-\,(\partial v/\partial p_i)/(\partial v/\partial m)$ e \emph{simplifique até o fim} --- só os expoentes de $p_i$ sobrevivem ao cancelamento, e é nesse cancelamento que o coeficiente da demanda aparece (ou some, se você atalhar). Avalie no ponto pedido e confira a dimensão: o número final é \emph{quantidade}, não razão de preços.}

\sub{Equação de Slutsky e taxonomia}
\[
\frac{\partial x_i^M}{\partial p_j}=\underbrace{\frac{\partial h_i}{\partial p_j}}_{\text{subst.}\,\le 0\ (i=j)}-\underbrace{x_j^M\,\frac{\partial x_i^M}{\partial m}}_{\text{renda}}.
\]
\emph{Normal/inferior}: sinal de $\partial x/\partial m$. \emph{Ordinário/Giffen}: sinal de $\partial x/\partial p$. Isolando o efeito-substituição no próprio preço:
\[
\frac{\partial h_1}{\partial p_1}=\frac{\partial x_1^M}{\partial p_1}+x_1^M\,\frac{\partial x_1^M}{\partial m}.
\]
O termo-renda entra \emph{com o sinal de $\partial x/\partial m$}: positivo num bem normal, negativo num bem inferior --- e $x_1^M\ge 0$ é só o peso.
\guia{Substitua na fórmula acima os três números que o enunciado fornece (efeito total, nível $x_1$, propensão $\partial x_1/\partial m$), \emph{respeitando o sinal} da propensão, e isole $\partial h_1/\partial p_1$. Depois cheque a soma de três peças: total $=$ substituição $+$ (efeito-renda \emph{sobre a demanda}, que é $-x_1\,\partial x_1/\partial m$) --- as três têm de fechar. Saiba percorrer as implicações entre as quatro categorias (normal/inferior $\times$ ordinário/Giffen) nas duas direções; o peso $x_j$ no termo-renda diz quando a renda domina.}

\sub{Elasticidades e classificação bruta}
$\varepsilon_{ij}=\dfrac{\partial x_i}{\partial p_j}\dfrac{p_j}{x_i}$ (adimensional --- \emph{não} confunda com a derivada),\ \ $\eta_i=\dfrac{\partial x_i}{\partial m}\dfrac{m}{x_i}$,\ \ $s_i=\dfrac{p_ix_i}{m}$. \emph{Agregação de Engel}: diferenciando $\sum_i p_i x_i^M=m$ em $m$ e ponderando,
\[
\sum_i s_i\,\eta_i=1,\qquad \textstyle\sum_i s_i=1.
\]
A média das elasticidades-renda, \emph{ponderada pelas participações}, é exatamente $1$. A elasticidade-preço cruzada classifica em termos \emph{brutos}: $\varepsilon_{ij}>0\Rightarrow$ substitutos brutos; $<0\Rightarrow$ complementos brutos; a versão \emph{líquida} (hicksiana) pode discordar do sinal, pois a bruta embute renda.
\guia{Tome a identidade $\sum_i s_i\eta_i=1$ como \emph{vínculo contábil} (vale por exaustão do orçamento, sem forma funcional): monte a média ponderada, imponha o valor de um $\eta_i$ dado e \emph{derive} o que sobra para os demais --- e só então julgue a afirmação proposta, distinguindo o que o vínculo força do que ele não determina.}

\sub{Bem-estar: VC, VE e excedente (alta de preço $p^0\to p^1$, renda $m$)}
Ordem de execução: (1) $v(p,m)$; (2) inverta para $e(p,\bar u)$; (3) avalie $u^0=v(p^0,m)$ e $u^1=v(p^1,m)$; (4) monte as variações. Pelo dispêndio:
\[
\text{VC}=e(p^1,u^0)-e(p^0,u^0)=e(p^1,u^0)-m,
\]
\[
\text{VE}=e(p^1,u^1)-e(p^0,u^1)=m-e(p^0,u^1),
\]
\[
\Delta EC=\int_{p_1^0}^{p_1^1}\! x_1^{M}(p_1,m)\,dp_1.
\]
VC usa o bem-estar \emph{inicial} $u^0$; VE usa o \emph{final} $u^1$ (mnemônico: VC$\leftrightarrow$inicial, VE$\leftrightarrow$final). A \emph{receita} de um imposto específico é $R_{\text{arr}}=t\cdot x_1$ \emph{na quantidade pós-imposto}; a folga entre a perda de bem-estar e essa receita é a semente do peso-morto.
\guia{Calcule VC e VE pela receita de 4 passos e, em paralelo, monte o lump-sum $L^{*}$ que iguala $v(p^0,m-L^{*})=u^1$; confirme \emph{você mesmo} se $L^{*}$ bate com uma das duas variações. Para localizar a fonte da perda de eficiência: tome a cesta escolhida sob o imposto, avalie-a aos preços \emph{verdadeiros}, e verifique se ela é viável sob um lump-sum de igual arrecadação --- então deixe o consumidor \emph{reotimizar} a esses preços e compare bem-estar. O peso-morto é a diferença entre \emph{qual} perda e \emph{qual} receita: identifique cada uma. Para qual classe de preferências VC$=$VE$=\Delta EC$?}

\bloco{Bloco 2 --- Equilíbrio Geral (Aulas 4--6)}

\sub{Caixa de Edgeworth (trocas, $2\times 2$)}
Factibilidade: $x^A+x^B=\omega^A+\omega^B$. \emph{Melhoria de Pareto}: ninguém piora, alguém melhora. No \emph{interior}, a CPO da eficiência é a tangência $\text{TMS}^A_{12}=\text{TMS}^B_{12}$. Curva de contrato: lugar geométrico desses pontos.

\sub{Equilíbrio walrasiano --- receita}
(1) Normalize (numerário): só preços \emph{relativos} importam (homog.\ grau 0). (2) Renda de dotação $m^i=p\cdot\omega^i$. (3) Demandas $x^i(p,m^i)$ --- p.\ ex.\ Cobb-Douglas $x_1^i=\alpha^i m^i/p_1$. (3.5) Monte o excesso de demanda de \emph{um} bem, $z_k(p)=\sum_i x_k^i(p,p\cdot\omega^i)-\sum_i\omega_k^i$, substituindo a renda de dotação \emph{dentro} da marshalliana antes de igualar a zero. (4) \emph{Market clearing}: $z_k(p)=0$ resolve $p^{*}$ (um basta, Lei de Walras $p\cdot z(p)=0$); o outro mercado fecha sozinho --- use-o como verificação. (5) Recupere as alocações.
\guia{Em que \emph{unidade} o preço relativo foi pedido --- bem 2 por bem 1, ou o inverso? Trocar o numerário substitui a resposta pelo seu recíproco (a armadilha clássica). Ao montar $z_k$, ponha a renda de dotação \emph{dentro} de cada demanda \emph{antes} de somar, e confirme no segundo mercado que a alocação fecha.}

\sub{Produção, Robinson Crusoé e FPP linear}
Firma competitiva: $\max p\,f(z)-p_1 z$, CPO \emph{valor do produto marginal $=$ custo do insumo}. Plano eficiente iguala a taxa marginal de transformação à razão de preços e à TMS: $\text{TMS}_{12}=\text{TMT}_{12}=p_1/p_2$. Para FPP \emph{curva} $T(x_1,x_2)=\bar c$, a TMT é $\dfrac{\partial T/\partial x_1}{\partial T/\partial x_2}$ (razão dos produtos marginais): substitua na fronteira e iguale a $p_1/p_2$. Com FPP \emph{linear} (capacidades $\bar x_1$ \emph{ou} $\bar x_2$), o custo de oportunidade de uma unidade do bem 1 é $\bar x_2/\bar x_1$, \emph{constante}, e fixa a taxa de autarquia; a \emph{vantagem comparativa} manda exportar o bem de menor custo de oportunidade, e o comércio beneficia ambos se o preço internacional cair \emph{entre} os dois custos.
\guia{Numa economia com produção e lucro distribuído, a renda do consumidor-acionista é $M(p)=p\cdot\omega+\pi^{*}(p)$: monte-a explicitamente. Sob retornos \emph{constantes}, calcule o lucro competitivo e conclua o que sobra de relevante na pergunta ``de quem é a firma''. Confira que cada extremo do intervalo de troca iguala um custo de autarquia.}

\sub{Teoremas do Bem-Estar}
\emph{1.\textsuperscript{o} TBE}: todo equilíbrio competitivo é Pareto-eficiente. Hipóteses (lista neutra): \emph{não-saciedade local}, agentes preço-tomadores, mercados completos, sem externalidades --- argumento puramente orçamentário. \emph{2.\textsuperscript{o} TBE}: toda alocação eficiente é sustentável como equilíbrio com transferências \emph{lump-sum}; hipóteses \emph{adicionais} de livro-texto: \emph{convexidade} de preferências/tecnologias e continuidade --- a sustentação passa por um \emph{hiperplano separador}.
\guia{Percorra o \emph{argumento} de cada teorema e marque, passo a passo, em qual linha cada hipótese é usada: no 1.\textsuperscript{o}, some as restrições orçamentárias de uma alocação que dominasse o equilíbrio e busque a contradição; no 2.\textsuperscript{o}, identifique o objeto geométrico que a construção do vetor de preços separa, e diga que propriedade desse objeto a separação exige. Não cite as hipóteses --- mostre onde elas entram.}

\sub{Incerteza: Arrow-Debreu e não-arbitragem}
Estados $s$, prob.\ $\pi_s$, consumo contingente $c_s$, $EU=\sum_s\pi_s v(c_s)$. CPO interior (Lagrangiano $\sum_s\pi_s v(c_s)+\lambda(W-\sum_s q_s c_s)$): $\dfrac{\pi_s v'(c_s)}{q_s}=\lambda$, donde $\dfrac{\pi_s v'(c_s)}{\pi_t v'(c_t)}=\dfrac{q_s}{q_t}$. Com $v=\ln c$: $c_s=\dfrac{\pi_s}{\lambda q_s}$. Riqueza $W=q\cdot\omega$. Preço \emph{justo} $q_s=\pi_s$.
\guia{Monte o Lagrangiano com a restrição $q\cdot c=q\cdot\omega$, derive $c_s=\pi_s/(\lambda q_s)$ e forme a \emph{razão} $c_1/c_2=(\pi_1 q_2)/(\pi_2 q_1)$ --- ela não depende de $\lambda$. Compare $q_s$ com $\pi_s$ caso a caso: deduza \emph{você} para que lado o consumo se inclina quando um estado está ``carregado'' ($q_s>\pi_s$), em vez de presumir suavização. Feche pela restrição orçamentária.}

\sub{Precificar por não-arbitragem e completude}
\emph{Título de Arrow} do estado $s$: paga 1 em $s$, 0 nos demais; preço $q_s$. Sequência: (1) para cada ativo negociado, $\text{preço}=\sum_s q_s\cdot\text{payoff}_s$; (2) resolva o sistema para $q_1,\dots,q_S$ (\emph{únicos} se completo); (3) precifique o alvo $P=\sum_s q_s\,y_s$; (4) cheque casando a carteira replicante. Mercado \emph{completo}: nº de ativos linearmente independentes $=S$ (span $=\mathbb{R}^S$); um seguro contra o estado $s$ é o próprio título de Arrow de $s$, de preço $q_s$.
\guia{Para qualquer coleção de ativos: monte a matriz de \emph{payoffs} por estado, conte os linearmente independentes, compare com $S$ e \emph{descreva por extenso} quais perfis de consumo contingente uma carteira alcança dentro do span (que componentes ela pode produzir, em que proporção, com que sinais). Só julgue afirmações de segurabilidade ou de transferência de risco \emph{depois} dessa conta, nunca antes.}

\bloco{Bloco 3 --- Externalidades e Bens Públicos (Aula 7)}

\sub{Externalidade e instrumento de Pigou}
\emph{Negativa de produção/consumo}: o privado iguala benefício marginal ao custo \emph{privado}, ignorando o dano $\Rightarrow$ produz demais; o ótimo iguala benefício marginal ao \emph{custo social} $=$ privado $+\,\text{DMg}$, e $t^{*}=\text{DMg}$ \emph{avaliado em $q^{*}$}. \emph{Positiva}: o ótimo iguala $\text{MPC}$ ao benefício marginal \emph{social} (privado $+\,\text{MEB}$), e o subsídio $s^{*}=\text{MEB}$ avaliado em $q^{*}$; a firma/agente passa a resolver com o preço efetivo corrigido.
\guia{Resolva os \emph{dois} equilíbrios (privado: BMg $=$ custo privado; social: inclua $\text{MEB}$ ou $\text{DMg}$) e anote a \emph{unidade} de cada número que aparece no caminho --- nível de output, sua variação, preço de demanda no ótimo, custo marginal --- e confira qual deles o item de instrumento corretivo de fato pede. Derive o sinal do instrumento a partir do sinal do efeito externo, não de memória.}

\sub{Recurso comum / tragédia dos comuns}
Sob \emph{acesso livre}, cada entrante embolsa o benefício \emph{médio} do recurso, $B(N)/N$, e entra enquanto esse médio cobrir o custo $\Rightarrow$ entrada cessa em $B(N)/N=c$. O planejador compara o benefício \emph{marginal} $B'(N)$ ao custo $\Rightarrow$ ótimo em $B'(N)=c$. A cunha médio-vs-marginal é a externalidade de congestionamento que o entrante não paga.
\guia{Para $B(N)$ dado, escreva \emph{separadamente} o benefício médio $B(N)/N$ e o marginal $B'(N)$, associe \emph{cada} condição (médio $=c$ / marginal $=c$) ao agente certo (entrante individual / planejador) \emph{antes} de resolver, e só então iguale ao custo. Derive a ordenação entre os dois níveis de $N$ da concavidade de $B$; quando coincidiriam?}

\sub{Bens públicos: Samuelson, agregação e cobertura}
Bem público puro: \emph{não-rival} e \emph{não-excludente}. Provisão eficiente agrega as valorações marginais:
\[
\sum_{i=1}^{I}\text{TMS}_i=\text{CMg}\ (=\text{TMT}).
\]
\emph{Lindahl}: cada agente $i$ enfrenta um preço-cota personalizado $\tau^i$ por unidade e demanda o \emph{mesmo} $G^{*}$; no equilíbrio $\tau^i=\text{TMS}^i(G^{*})$ e $\sum_i\tau^i=\text{CMg}$. \emph{Provisão voluntária}: cada um iguala a \emph{própria} TMS ao custo, ignorando o benefício alheio (carona, subprovisão).
\guia{Para agregar valorações, decida primeiro \emph{o que} se soma e \emph{o que} se mantém fixo, \emph{derivando} essa decisão da rivalidade de cada bem caso a caso (no público puro, todos consomem o mesmo nível ao mesmo tempo) --- e confira a altura agregada num ponto numérico recompondo as disposições individuais. Na provisão voluntária, derive a melhor-resposta de cada agente a partir da CPO da \emph{sua} utilidade (tratando o canto $g^i\ge 0$ explicitamente), caracterize o equilíbrio e compare o total provido com o $G^{*}$ de Samuelson.}

\sub{Externalidade que cruza fronteiras / jurisdição}
O instrumento corretivo tem \emph{duas} dimensões independentes: a \emph{alíquota} (igualar o custo efetivo ao dano marginal, avaliado no ótimo) e a \emph{cobertura} (o conjunto de fontes e de danos que o instrumento alcança).
\guia{Diante de um instrumento corretivo proposto, faça \emph{duas} checagens separadas antes de julgar otimalidade: (i) a alíquota iguala o quê a quê, avaliada onde? (ii) liste o conjunto de danos que entram no cálculo de quem fixa a alíquota e o conjunto de danos sociais totais, e compare-os. Conclua só depois das duas checagens.}

\bloco{Bloco 4 --- Assimetria Informacional e Sinalização (Aulas 8--9)}

\sub{Mapa do território}
\emph{Informação oculta} (característica, \emph{antes} do contrato) $\rightarrow$ seleção adversa. \emph{Ação oculta} (esforço, \emph{depois}) $\rightarrow$ risco moral. \emph{Verificação} \emph{ex post} é um terceiro eixo. Informado emite \emph{sinal}; não-informado desenha \emph{menu} (screening).

\sub{Seleção adversa (Akerlof) --- receita}
A um preço $p$, vende quem tem reserva $\le p$; com qualidade $q\sim U[0,\bar q]$ e reserva $r\,q$, vendem os $q\le p/r$, e a média \emph{condicional} é $\tfrac12\,(p/r)$. Comprador neutro paga $p$ só se sua valoração esperada $k\cdot\mathbb{E}[q\mid\text{vende}]$ cobrir $p$. Equilíbrio: ponto fixo entre $p$ e quem vende; pode haver \emph{unraveling} (colapso).
\guia{Monte a média condicional dos que vendem \emph{aplicando} o fator de reserva ($q\le p/r$, não $q\le p$) e o viés de truncamento $\tfrac12$; imponha a desigualdade de aceitação do comprador, cancele $p$ e isole o parâmetro de prêmio. O limiar é um \emph{recíproco} de fração --- derive-o, não leia coeficientes soltos do enunciado.}

\sub{Sinalização (Spence) e garantia como sinal}
Sinal custoso com \emph{single-crossing}: custo marginal do sinal \emph{menor} para o tipo melhor. \emph{Separador}: o sinal é caro o bastante para o tipo pior \emph{não} imitar. Uma \emph{garantia} de valor $B$ acionada com probabilidade $p_i$ custa, em esperança, $p_i B$ --- maior para o tipo pior ($p_F>p_R$). As duas restrições de incentivo:
\[
\underbrace{p_F\,B\ \ge\ \Delta}_{\text{pior não imita}}\qquad\text{e}\qquad \underbrace{p_R\,B\ \le\ \Delta}_{\text{melhor sinaliza}},
\]
com $\Delta$ o prêmio de parecer bom.
\guia{Monte as duas desigualdades com os dados, resolva \emph{cada uma} para $B$ e examine o intervalo resultante por inteiro antes de responder qualquer coisa sobre sinal ``mínimo'' ou ``máximo''. Pergunte-se também se o sinal precisa elevar produtividade real para separar.}

\sub{Risco moral e contrato fixo vs.\ bônus}
Esforço não observável $\Rightarrow$ contrato condiciona só no resultado. Trade-off: \emph{seguro} (proteger o avesso) vs.\ \emph{incentivo} (expô-lo ao resultado). \emph{IC de esforço} (avesso): $\sum_s \pi_s^{e}v(w_s)-c(e)\ge\sum_s \pi_s^{0}v(w_s)$. Agente \emph{neutro}, contrato base fixa $+$ fração $\beta$ do resultado: a IC vira linear, $\beta\,\mathbb{E}[R\mid e{=}1]-c\ge\beta\,\mathbb{E}[R\mid e{=}0]\Leftrightarrow\beta\ge c/\Delta_R$, com $\Delta_R$ o incremento esperado. A base fixa cancela dos dois lados.
\guia{Monte $\Delta_R$ a partir dos \emph{dois} cenários de esforço --- escreva as duas esperanças por extenso, com todos os estados --- antes de dividir. Identifique por que a base fixa não move o esforço e franquia/$\beta$ movem. First-best (esforço observável) vs.\ second-best: a distorção é o custo da informação.}

\sub{Screening de monopólio (menu de 2.\textsuperscript{o} grau)}
Tipos $\theta_L<\theta_H$, valor $\theta q$, custo $C(q)$, excedente $\theta q-T$. \emph{First-best} (tipo observável): $q_i^{FB}$ eficiente ($\theta_i=C'(q_i)$), IR vincula, $T_i=\theta_i q_i$ extrai tudo --- \emph{sem renda informacional}. \emph{Second-best} (menu): no ótimo, duas das quatro restrições valem com igualdade; o tipo $H$ retém uma \emph{renda informacional} (o excedente que teria imitando o pacote do $L$) e a quantidade de um dos tipos é distorcida em relação ao first-best.
\guia{Monte as quatro restrições (duas IR, duas IC), identifique \emph{derivando} quais vinculam, e extraia a renda do $H$ da restrição ativa --- ela sai proporcional a uma quantidade do menu; descubra qual e o que isso implica para o desenho. Para o custo da assimetria (queda FB$\to$SB), liste as fontes candidatas, calcule cada parcela e confira que somam o total. Uma \emph{taxa de adesão} lump-sum entra em quais restrições? Refaça a comparação de imitação \emph{com} a taxa incluída e conclua você mesmo o efeito.}

\sub{Matching (aceitação diferida, Gale-Shapley)}
Pareamento \emph{estável}: sem \emph{par bloqueante} (dois agentes não-pareados entre si que se preferem aos pares atuais). \emph{Aceitação diferida}: propostas iterativas com aceitação provisória; termina sempre em pareamento estável; o lado \emph{proponente} obtém seu \emph{melhor} pareamento estável.
\guia{Para testar uma alocação proposta, tome cada par não-pareado e cheque \emph{ambas} as direções de preferência (os dois lados têm de se preferir ao par atual) --- e classifique a alocação só depois de varrer os pares. Para as propriedades de incentivo, derive-as da otimalidade-do-proponente: pergunte, para cada lado, o que ele ganha reportando preferências distintas das verdadeiras, e teste num micro-exemplo $3\times 3$ antes da prova.}
