\exercicio{1}{Mercado completo: equilĂ­brio AD, preĂ§os de Arrow e implementaĂ§ĂŁo Radner (com \emph{risco agregado})}

\begin{setupbox}
\begin{itemize}[leftmargin=1.3em,itemsep=2pt,topsep=2pt]
  \item 1 bem por estado, \textbf{2 estados}: $s=1$ (recessĂŁo), $s=2$ (expansĂŁo). $I=2$ agentes.
  \item CrenĂ§as $\pi=(1/2,\,1/2)$; utilidade \textbf{log}: $u^i=\tfrac12\ln c_1+\tfrac12\ln c_2$.
  \item DotaĂ§Ăµes $\omega^A=(60,\,60)$, $\omega^B=(40,\,240)$.
  \item Agregado $\bar\omega=(100,\,300)$ --- \textbf{arriscado} (a expansĂŁo triplica o estado~1).
  \item Mercado de Radner: tĂ­tulo $D^{\mathrm{bond}}=(1,1)$ e aĂ§ĂŁo $D^{\mathrm{aĂ§ĂŁo}}=(1,3)$, com
        matriz de payoffs $A=\begin{pmatrix}1&1\\1&3\end{pmatrix}$.
\end{itemize}
\textbf{Perguntas.} (a) preĂ§os de Arrow $p$ (normalize $p_1+p_2=1$); (b) alocaĂ§ĂŁo de equilĂ­brio AD;
(c) preĂ§os dos ativos por nĂŁo-arbitragem; (d) portfĂłlio Radner $\theta^A$ que implementa $c^A$ + restriĂ§ĂŁo em $t=0$;
(e) \emph{clearing} do mercado de ativos.
\end{setupbox}

\passo{Passo 0 --- Demanda log (deduza no quadro).}
$\max\sum_s\pi_s\ln c_s$ sujeito a $\sum_s p_s c_s=m^i$. CPO: $\pi_s/c_s=\lambda p_s\Rightarrow c_s=\pi_s/(\lambda p_s)$.
Na restriĂ§ĂŁo, $\sum_s p_s c_s=\tfrac1\lambda\sum_s\pi_s=\tfrac1\lambda=m^i$, logo
\[
  \boxed{\,c^i_s=\frac{\pi_s\,m^i}{p_s}\,},\qquad m^i=p\cdot\omega^i.
\]
O consumidor gasta a fraĂ§ĂŁo $\pi_s$ da riqueza no \emph{claim} do estado $s$ (Cobb--Douglas/log).

\passo{Passo 1 --- PreĂ§os de Arrow (a).}
\emph{Clearing} no estado $s$: $\sum_i c^i_s=\bar\omega_s$, ou seja
\[
  \frac{\pi_s}{p_s}\underbrace{(m^A+m^B)}_{=\,M\,=\,p\cdot\bar\omega}=\bar\omega_s
  \;\Longrightarrow\;
  p_s=\frac{\pi_s\,M}{\bar\omega_s}\;\propto\;\frac{\pi_s}{\bar\omega_s}.
\]
Com $\pi=(1/2,1/2)$: $p_s\propto 1/\bar\omega_s$, entĂŁo $p_1:p_2=\tfrac1{100}:\tfrac1{300}=3:1$. Logo
\[
  \boxed{\,p=(3/4,\;1/4)\,}.
\]

\begin{intuibox}
\textbf{IntuiĂ§ĂŁo (SDF / consumption-CAPM).} $p_s/\pi_s=M/\bar\omega_s$ Ă© \emph{maior no estado escasso}
(recessĂŁo). Consumir na recessĂŁo Ă© \emph{caro} porque todo mundo estĂˇ pobre lĂˇ. O preĂ§o de Arrow por unidade
de probabilidade \emph{Ă©} o fator de desconto estocĂˇstico.
\end{intuibox}

\passo{Passo 2 --- AlocaĂ§ĂŁo de equilĂ­brio (b).}
Riquezas: $m^A=\tfrac34(60)+\tfrac14(60)=45+15=60$ e $m^B=\tfrac34(40)+\tfrac14(240)=30+60=90$.
(Cheque: $M=60+90=150=p\cdot\bar\omega=\tfrac34(100)+\tfrac14(300)=75+75=150$ \ck.)
\[
  c^A=\Big(\tfrac{(1/2)\,60}{3/4},\;\tfrac{(1/2)\,60}{1/4}\Big)=(40,\,120),
  \qquad
  c^B=\Big(\tfrac{(1/2)\,90}{3/4},\;\tfrac{(1/2)\,90}{1/4}\Big)=(60,\,180).
\]
\emph{Clearing}: $(40{+}60,\;120{+}180)=(100,300)=\bar\omega$ \ck. \quad
Trocas: $A:(-20,+60)$, $B:(+20,-60)$, soma nula \ck.

\begin{intuibox}
\textbf{Ponto pedagĂłgico de ouro.} $A$ comeĂ§ou \textbf{seguro} $(60,60)$ e termina \textbf{arriscado} $(40,120)$.
Com risco agregado \emph{ninguĂ©m} se segura totalmente: cada agente consome uma \textbf{fraĂ§ĂŁo fixa do agregado}
($c^A/\bar\omega=(0{,}4;\,0{,}4)$, $c^B/\bar\omega=(0{,}6;\,0{,}6)$), igual Ă  sua participaĂ§ĂŁo na riqueza
($m^A/M=0{,}4$). Ă‰ o \textbf{princĂ­pio da mutualidade de Borch}.
\end{intuibox}

\passo{Passo 3 --- PreĂ§os dos ativos (c), nĂŁo-arbitragem $q_j=\sum_s p_s A_{sj}$.}
\[
  q_{\mathrm{bond}}=\tfrac34(1)+\tfrac14(1)=1,
  \qquad
  q_{\mathrm{aĂ§ĂŁo}}=\tfrac34(1)+\tfrac14(3)=\tfrac34+\tfrac34=\tfrac32=1{,}5.
\]
\emph{ExtensĂŁo.} Retorno esperado da aĂ§ĂŁo $=\tfrac12(1)+\tfrac12(3)=2$; retorno bruto esperado $=2/1{,}5\approx1{,}33>R_f=1$.
O \textbf{prĂŞmio de risco Ă© positivo} porque a aĂ§ĂŁo paga mais na expansĂŁo (estado de Arrow barato): covaria com a
riqueza agregada e por isso exige prĂŞmio.

\passo{Passo 4 --- PortfĂłlio Radner de $A$ (d).}
Troca lĂ­quida $c^A-\omega^A=(40-60,\;120-60)=(-20,\,60)$. Resolva $A\theta^A=(-20,60)$:
\[
  \begin{cases}
    \theta_{\mathrm{bond}}+\theta_{\mathrm{aĂ§ĂŁo}}=-20 & (\text{estado }1)\\[2pt]
    \theta_{\mathrm{bond}}+3\theta_{\mathrm{aĂ§ĂŁo}}=60 & (\text{estado }2)
  \end{cases}
  \;\Rightarrow\;
  2\theta_{\mathrm{aĂ§ĂŁo}}=80\Rightarrow\theta_{\mathrm{aĂ§ĂŁo}}=40,\;\;\theta_{\mathrm{bond}}=-60.
\]
\[
  \boxed{\,\theta^A=(-60,\,40)\,}\quad(\text{vende 60 tĂ­tulos, compra 40 aĂ§Ăµes}).
\]
\textbf{RestriĂ§ĂŁo em $t=0$:} $q\cdot\theta^A=1(-60)+1{,}5(40)=-60+60=0$ \ck{} --- algebricamente equivalente Ă 
Lei de Walras AD de $A$.

\passo{Passo 5 --- \emph{Clearing} de ativos (e).}
Por simetria, $A\theta^B=(20,-60)\Rightarrow\theta^B=(60,-40)$, com $q\cdot\theta^B=60-60=0$ \ck{} e
$\theta^A+\theta^B=(0,0)$ \ck. \;
\textbf{Fechamento:} a sequĂŞncia Radner reproduz \emph{exatamente} a alocaĂ§ĂŁo AD $(40,120)/(60,180)$ --- mesma
economia, instituiĂ§ĂŁo diferente, porque $\operatorname{rank}(A)=2=|S|$.

\begin{armadbox}
\textbf{Armadilhas.} (i) normalizar $p$ por $\pi$ esquecendo $\bar\omega$; (ii) confundir a troca lĂ­quida $c-\omega$
com o prĂłprio $\theta$ (esquecer o $A^{-1}$); (iii) achar que $A$ ``deveria'' continuar seguro.
\end{armadbox}

\exercicio{2}{Mercado \textbf{incompleto}: span, seguro inviĂˇvel e completar o mercado (Hart/GP)}

\begin{setupbox}
\begin{itemize}[leftmargin=1.3em,itemsep=2pt,topsep=2pt]
  \item 1 bem por estado, \textbf{3 estados}, $I=2$, $\pi=(1/3,1/3,1/3)$, utilidade log.
  \item DotaĂ§Ăµes $\omega^A=(100,\,40,\,40)$, $\omega^B=(20,\,80,\,80)$. Agregado $\bar\omega=(120,120,120)$ --- \textbf{livre de risco}.
  \item \textbf{CenĂˇrio 1:} tĂ­tulo $(1,1,1)$ + aĂ§ĂŁo $(0,1,2)$, $A_2=\begin{pmatrix}1&0\\1&1\\1&2\end{pmatrix}$.
  \item \textbf{CenĂˇrio 2:} acrescenta opĂ§ĂŁo $(0,0,1)$, $A_3=\begin{pmatrix}1&0&0\\1&1&0\\1&2&1\end{pmatrix}$.
\end{itemize}
\textbf{Perguntas.} (a) rank/completude no CenĂˇrio 1; (b) o seguro total de $A$ estĂˇ no span?
(c) preĂ§os de Arrow do AD ideal; (d) CenĂˇrio 2: complete o mercado e ache $\theta^A$ + restriĂ§ĂŁo $t=0$;
(e) bem-estar --- completar ajuda ou prejudica aqui?
\end{setupbox}

\passo{Passo 1 --- Rank e completude (a).}
As colunas $(1,1,1)^\top$ e $(0,1,2)^\top$ nĂŁo sĂŁo mĂşltiplas $\Rightarrow$ independentes $\Rightarrow$
$\operatorname{rank}(A_2)=2<3=|S|$. \textbf{Mercado incompleto}; o span Ă© um plano 2D em $\R^3$.

\passo{Passo 2 --- AlocaĂ§ĂŁo ideal AD e teste de span (b).}
Agregado livre de risco + log + crenĂ§as uniformes $\Rightarrow$ \textbf{seguro total}. A preĂ§os justos
$p=(1/3,1/3,1/3)$, $m^A=\tfrac13(100{+}40{+}40)=60\Rightarrow c^A=(60,60,60)$. TransferĂŞncia necessĂˇria:
\[
  c^A-\omega^A=(60-100,\;60-40,\;60-40)=(-40,\,20,\,20).
\]
EstĂˇ no span? Resolva $a(1,1,1)+b(0,1,2)=(-40,20,20)$:
\[
  s_1:\;a=-40;\qquad s_2:\;a+b=20\Rightarrow b=60;\qquad s_3:\;a+2b=-40+120=80\neq20.
\]
\textbf{Inconsistente $\Rightarrow(-40,20,20)\notin$ span: $A$ nĂŁo consegue se segurar totalmente.}

\begin{intuibox}
\textbf{IntuiĂ§ĂŁo.} Os instrumentos sĂł geram transferĂŞncias nas direĂ§Ăµes ``uniforme'' $(1,1,1)$ e ``inclinada
pro estado 3'' $(0,1,2)$. $A$ precisaria \emph{ganhar igual nos estados 2 e 3} cortando o estado 1 --- mas a aĂ§ĂŁo
paga \emph{mais no 3 que no 2}, e nenhuma combinaĂ§ĂŁo entrega isso. O equilĂ­brio Radner \emph{existe}
(Magill--Quinzii), Ă© \textbf{constrained Pareto-eficiente} (Ăłtimo dentro do span), mas \textbf{Pareto-inferior}
ao AD ideal --- Hart (1975).
\end{intuibox}

\passo{Passo 3 --- PreĂ§os de Arrow do AD ideal (c).}
Agregado constante + crenĂ§as uniformes $\Rightarrow$ preĂ§os \textbf{atuarialmente justos} $p=(1/3,1/3,1/3)$
(mesmo argumento $p_s\propto\pi_s/\bar\omega_s$ do Ex.~1). NĂŁo-arbitragem:
\[
  q_{\mathrm{bond}}=\tfrac13(1{+}1{+}1)=1,\quad
  q_{\mathrm{aĂ§ĂŁo}}=\tfrac13(0{+}1{+}2)=1,\quad
  q_{\mathrm{opĂ§ĂŁo}}=\tfrac13(0{+}0{+}1)=\tfrac13.
\]

\passo{Passo 4 --- CenĂˇrio 2: completar e implementar (d).}
$\det A_3=1$ (triangular inferior, diagonal $1{\cdot}1{\cdot}1$) $\neq0\Rightarrow\operatorname{rank}=3\Rightarrow$
\textbf{mercado completo}. Agora $(-40,20,20)$ Ă© alcanĂ§Ăˇvel. Resolva $A_3\,\theta^A=(-40,20,20)$:
\[
  s_1:\;a=-40;\quad s_2:\;a+b=20\Rightarrow b=60;\quad s_3:\;a+2b+c=20\Rightarrow-40+120+c=20\Rightarrow c=-60.
\]
\[
  \boxed{\,\theta^A=(-40,\,60,\,-60)\,}\quad(\text{vende 40 tĂ­tulos, compra 60 aĂ§Ăµes, vende 60 opĂ§Ăµes}).
\]
\textbf{RestriĂ§ĂŁo em $t=0$:} $q\cdot\theta^A=1(-40)+1(60)+\tfrac13(-60)=-40+60-20=0$ \ck.

\passo{Passo 5 --- Bem-estar (e).}
Como o agregado Ă© \textbf{livre de risco}, o seguro total $(60,60,60)$ Ă© viĂˇvel para ambos no CenĂˇrio 2 e
estritamente preferido (aversĂŁo a risco) a qualquer alocaĂ§ĂŁo nĂŁo-segurada do CenĂˇrio 1. Logo
\textbf{CenĂˇrio 2 $\succ_{\text{Pareto}}$ CenĂˇrio 1}: aqui, completar o mercado \textbf{melhora} todo mundo.

\begin{intuibox}
\textbf{Contraste crucial (GP 1986).} Isso sĂł vale porque o agregado Ă© \emph{livre de risco} (alocaĂ§ĂŁo ideal
simĂ©trica, ambos ganham). Se o agregado fosse \emph{arriscado} (como no Ex.~1), completar redistribuiria risco
e \emph{poderia prejudicar} algum agente --- ``mais ativos'' nĂŁo Ă© trivialmente bom. Ă‰ a fronteira entre Hart
(ineficiĂŞncia genĂ©rica) e Geanakoplos--Polemarchakis (completar pode piorar).
\end{intuibox}

\begin{armadbox}
\textbf{Armadilhas.} (i) testar sĂł 2 das 3 equaĂ§Ăµes no span e concluir errado; (ii) confundir
``ineficiente'' com ``inexistente'' (o equilĂ­brio Radner \emph{existe}); (iii) generalizar ``completar sempre
melhora'' --- exatamente o que GP derruba.
\end{armadbox}
