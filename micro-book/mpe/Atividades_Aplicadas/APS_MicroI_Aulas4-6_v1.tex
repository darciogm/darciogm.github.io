    \noindent\textbf{FormulĂĄrio e notaĂ§ĂŁo permitidos}
    \vspace{0.3em}

    \textbf{EquilĂ­brio competitivo (EC) em troca pura.} Em economia $I \times L$ com preferĂŞncias $\succeq^i$, dotaĂ§Ăľes $\omega^i \in \mathbb{R}^L_{++}$ e preĂ§o $p \in \mathbb{R}^L_+$:
    \[
        x^i(p) \in \arg\max\, u^i(x^i)\ \text{s.a.}\ p\cdot x^i \leq p\cdot \omega^i,\qquad
        z(p) = \sum_i (x^i(p) - \omega^i),\qquad p^*\cdot z(p^*) = 0\ \text{(Walras).}
    \]
    EC: $z(p^*) = 0$. \textbf{Taxa Marginal de SubstituiĂ§ĂŁo:} $\TMS^i_{\ell, k}(x^i) = (\partial u^i/\partial x^i_\ell)/(\partial u^i/\partial x^i_k)$.

    \textbf{ProduĂ§ĂŁo e PPF em economia $1+1$ bens / 1 fator.} Tecnologias $q_h = f_h(L_h)$, $q_t = f_t(L_t)$, restriĂ§ĂŁo $L_h + L_t = \bar L$. \textbf{Taxa Marginal de TransformaĂ§ĂŁo:} $\TMT_{h,t} = -dq_t/dq_h\big|_{\text{PPF}} = \mathrm{PMa}_L^{h}/\mathrm{PMa}_L^{t}$. EficiĂŞncia produtiva: $\TMT = p_h/p_t$.

    \textbf{Robinson Crusoe descentralizado.} Firma: $\pi(p, w) = \max_L p\, f(L) - w L$, CPO $p f'(L) = w$. Consumidor: renda $m = w \bar L + \pi^*$, $\max u(q, \ell)$ s.a.\ $p q + w \ell \leq m$. EC: $\TMS_{\ell, q} = w/p = \TMT_{L, q}$.

    \textbf{Arrow-Debreu (AD) com bens contingentes.} Estados $S = \{1, \ldots, |S|\}$, bens contingentes $(\ell, s)$, espaĂ§o $\mathbb{R}^{LS}$. PreĂ§o $p \in \mathbb{R}^{LS}_+$, dotaĂ§ĂŁo $\omega^i \in \mathbb{R}^{LS}_+$, EU: $u^i(x^i) = \sum_s \pi^i_s\, v^i(x^i_s)$ (caso $L = 1$). UMP em $\mathbb{R}^{LS}$:
    \[
        \max u^i(x^i)\ \text{s.a.}\ \sum_{(\ell,s)} p_{\ell, s}\, x^i_{\ell, s} \leq \sum_{(\ell,s)} p_{\ell, s}\, \omega^i_{\ell, s}.
    \]
    \textbf{EquaĂ§ĂŁo fundamental do SDF} (sob EU, $L = 1$, Ăłtimo interior):
    \[
        \boxed{\ \frac{p^*_s}{p^*_{s'}} \;=\; \frac{\pi^i_s\, (v^i)'(x^{i*}_s)}{\pi^i_{s'}\, (v^i)'(x^{i*}_{s'})}\quad \forall i.\ }
    \]
    Stochastic Discount Factor de mercado: $m_s \equiv p^*_s / \pi_s$ (sob crenĂ§as de consenso).

    \textbf{Radner sequencial.} $N$ ativos com matriz de payoffs $A \in \mathbb{R}^{|S| \times N}$, $A_{sj}$ ĂŠ o payoff do ativo $j$ no estado $s$. PreĂ§os de ativos $q \in \mathbb{R}^N$. EquilĂ­brio Radner com agregaĂ§ĂŁo $\sum_i \theta^i = 0$ e mercado spot estado-a-estado.

    \textbf{EquivalĂŞncia AD $\leftrightarrow$ Radner} (Arrow 1953; Radner 1972): vale se $\rank(A) = |S|$ (mercado completo via portfolio). NĂŁo-arbitragem: $q^*_j = \sum_s p^*_s\, A_{sj}$. RĂŠplica de aĂ§ĂŁo Arrow $e_s$: $\theta = A^{-1} e_s$ (existe sse $\rank(A) = |S|$).

    \textbf{EquaĂ§ĂŁo de Euler intertemporal.} Especializando $s = t$ com fator de desconto $\beta$ e taxa bruta de juros $R$:
    \[
        u'(c_t) = \beta\, R\, u'(c_{t+1}),\qquad 1 = E^\pi_t[m_{t+1}\, R_{t+1}],\quad m_{t+1} = \beta\, u'(c_{t+1})/u'(c_t).
    \]

\begin{contextbox}[Contexto: setor elĂŠtrico brasileiro e risco hidrolĂłgico]
\small
A matriz elĂŠtrica brasileira ĂŠ majoritariamente \textbf{hidrelĂŠtrica} ($\sim$60\% da geraĂ§ĂŁo em 2023, conforme \emph{Boletim Mensal de OperaĂ§ĂŁo} do ONS), complementada por \textbf{termelĂŠtrica} (gĂĄs natural, Ăłleo, carvĂŁo, nuclear), eĂłlica e solar. A geraĂ§ĂŁo hidrelĂŠtrica depende de hidrologia: em anos de chuvas favorĂĄveis, os reservatĂłrios enchem e a geraĂ§ĂŁo hidro entrega \emph{mais} energia que sua \textbf{Garantia FĂ­sica} (a quantidade firme contratada); em anos secos, entrega \emph{menos}.

O instrumento que sintetiza esse risco ĂŠ o \textbf{GSF} (\emph{Generation Scaling Factor}), razĂŁo entre a geraĂ§ĂŁo hidro efetiva e o total contratado pelo bloco hidro como Garantia FĂ­sica. Quando $\GSF < 1$, os geradores hidro sĂŁo, em conjunto, \emph{deficitĂĄrios}: tĂŞm que cobrir a diferenĂ§a comprando energia no mercado de curto prazo, ao \textbf{PLD} (PreĂ§o de LiquidaĂ§ĂŁo das DiferenĂ§as), divulgado semanalmente pela \textbf{CCEE} (CĂ˘mara de ComercializaĂ§ĂŁo de Energia ElĂŠtrica). O \textbf{MRE} (Mecanismo de RealocaĂ§ĂŁo de Energia), instituĂ­do pela Lei 9.648/1998, mutualiza esse risco \emph{entre} os geradores hidro -- os superavitĂĄrios ``emprestam'' energia aos deficitĂĄrios a preĂ§o regulado (TEO, Tarifa de Energia de OtimizaĂ§ĂŁo).

A \textbf{crise hĂ­drica de 2014--2015} (e novamente de 2021) levou o $\GSF$ mĂŠdio a nĂ­veis historicamente baixos ($\sim$0,7 em 2014--15 segundo dados publicados pela CCEE), gerando volume de exposiĂ§ĂŁo financeira que colapsou o equilĂ­brio do MRE. A \textbf{Lei 14.052/2020} repactuou esse passivo: parte do risco hidrolĂłgico foi reclassificada como ``risco nĂŁo-gerenciĂĄvel'' e transferida ao consumidor cativo via prorrogaĂ§ĂŁo de outorgas, reduzindo a exposiĂ§ĂŁo residual dos geradores hidro. A discussĂŁo regulatĂłria pĂłs-2020 -- conduzida pela \textbf{ANEEL} e pela CCEE -- gira em torno de quanto desse risco ĂŠ ``mutualizĂĄvel'' entre geradores (\emph{intra-bloco hidro}) versus quanto exige \emph{spanning} via instrumentos financeiros adicionais (CCEAR -- Contrato de ComercializaĂ§ĂŁo de Energia no Ambiente Regulado, contratos a termo, derivativos sobre PLD).

Esta APS modela uma economia estilizada com dois agentes (gerador hidro $H$ e consumidor industrial / tĂŠrmica $T$) e dois bens (energia firme e capital de giro/serviĂ§os), em trĂŞs nĂ­veis de complexidade. VocĂŞ (i) calcula o equilĂ­brio competitivo \emph{sem incerteza} (Q1, Aula 4) -- com produĂ§ĂŁo descentralizada hidro/tĂŠrmica via Robinson Crusoe; (ii) introduz dois estados da natureza (favorĂĄvel vs.\ seca) e calcula o equilĂ­brio Arrow-Debreu canĂ´nico, mostrando o \emph{seguro completo} entre $H$ avesso e $T$ neutro (Q2, Aula 5); (iii) implementa o equilĂ­brio de Q2 via Radner sequencial com 2 ativos (CCEAR-bond + aĂ§ĂŁo contingente hidro), depois amplia para 3 estados (com seca extrema) e diagnostica a \textbf{incompletude} de mercado e a Pareto-ineficiĂŞncia genĂŠrica de Hart (1975) (Q3, Aula 6); finalmente (iv) recomenda Ă  ANEEL/MME um redesenho do MRE pĂłs-Lei 14.052/2020 (Q4).

\begin{questions}
\question \textbf{EquilĂ­brio Geral em troca pura e produĂ§ĂŁo no setor elĂŠtrico} (25 pontos).

\noindent\emph{Parte (a)--(b): troca pura entre $H$ e $T$.} Suponha (em \emph{ausĂŞncia de incerteza}, calibraĂ§ĂŁo base de Q1) que o gerador hidro $H$ tem dotaĂ§ĂŁo $\omega^H = (\,3,\, 1\,)$ -- 3 unidades de energia, 1 de capital -- e o consumidor industrial $T$ tem dotaĂ§ĂŁo $\omega^T = (\,1,\, 3\,)$. PreferĂŞncias Cobb-Douglas:
\[
    u^H(x_1, x_2) = x_1^{1/3}\, x_2^{2/3},\qquad
    u^T(x_1, x_2) = x_1^{2/3}\, x_2^{1/3}.
\]
$T$ valoriza relativamente mais energia (parĂ˘metro $2/3$ no bem 1) -- ĂŠ o consumidor industrial, intensivo em energia. $H$ valoriza relativamente mais capital de giro (precisa financiar manutenĂ§ĂŁo dos reservatĂłrios). NumerĂĄrio: $p_2 = 1$.

\begin{parts}
\part[6] Resolva o equilĂ­brio competitivo desta economia de troca pura.

\textbf{(i)} Derive as demandas marshallianas $x^H_1(p_1)$ e $x^T_1(p_1)$ em funĂ§ĂŁo do preĂ§o relativo $p_1$ (numerĂĄrio $p_2 = 1$).

\textbf{(ii)} Aplique a condiĂ§ĂŁo de fechamento do mercado 1 ($x^H_1 + x^T_1 = \omega^H_1 + \omega^T_1 = 4$) para obter $p_1^*$. Verifique a Lei de Walras explicitamente computando $z_2(p^*)$.

\textbf{(iii)} Calcule a alocaĂ§ĂŁo de equilĂ­brio $(x^{H*}, x^{T*})$ e verifique que $\TMS^H = \TMS^T = p_1^*/p_2^*$.

\begin{solution}
\textbf{(i) Demandas Marshallianas (Cobb-Douglas).} Em CD com pesos $(\alpha_1, \alpha_2)$ e preĂ§os $(p_1, p_2)$, $x_\ell = \alpha_\ell\, m/p_\ell$. Renda: $m^i = p_1 \omega^i_1 + p_2 \omega^i_2 = p_1 \omega^i_1 + \omega^i_2$ (numerĂĄrio).
\[
    x^H_1(p_1) = \tfrac{1}{3}\cdot \frac{m^H}{p_1} = \tfrac{1}{3}\cdot \frac{3 p_1 + 1}{p_1} = 1 + \tfrac{1}{3 p_1};
\]
\[
    x^T_1(p_1) = \tfrac{2}{3}\cdot \frac{m^T}{p_1} = \tfrac{2}{3}\cdot \frac{p_1 + 3}{p_1} = \tfrac{2}{3} + \tfrac{2}{p_1}.
\]

\textbf{(ii) Fechamento do mercado 1.}
\[
    x^H_1 + x^T_1 = 1 + \tfrac{1}{3 p_1} + \tfrac{2}{3} + \tfrac{2}{p_1} = \tfrac{5}{3} + \tfrac{7}{3 p_1} = 4
    \;\Longrightarrow\; \tfrac{7}{3 p_1} = \tfrac{7}{3}
    \;\Longrightarrow\; \boxed{p_1^* = 1.}
\]

\textbf{VerificaĂ§ĂŁo Walras (mercado 2).} Com $p_1^* = 1$: $m^H = 3 + 1 = 4$, $m^T = 1 + 3 = 4$. Demanda do bem 2:
\[
    x^H_2 = \tfrac{2}{3}\cdot \tfrac{m^H}{p_2} = \tfrac{8}{3};\quad x^T_2 = \tfrac{1}{3}\cdot \tfrac{m^T}{p_2} = \tfrac{4}{3};\quad x^H_2 + x^T_2 = 4 = \omega^H_2 + \omega^T_2.\ \checkmark
\]
Equivalentemente, $z_2(p^*) = (8/3 + 4/3) - 4 = 0$ e $p^* \cdot z(p^*) = 1\cdot 0 + 1\cdot 0 = 0$. A Lei de Walras vale identicamente; em $L = 2$, fechar 1 mercado fecha o outro.

\textbf{(iii) AlocaĂ§ĂŁo e $\TMS$.}
\[
    x^{H*} = \left(1 + \tfrac{1}{3},\, \tfrac{8}{3}\right) = \left(\tfrac{4}{3},\, \tfrac{8}{3}\right),\quad
    x^{T*} = \left(\tfrac{2}{3} + 2,\, \tfrac{4}{3}\right) = \left(\tfrac{8}{3},\, \tfrac{4}{3}\right).
\]
$\TMS^H_{1,2} = (1/3)/(2/3) \cdot x^H_2/x^H_1 = (1/2)\cdot (8/3)/(4/3) = (1/2)\cdot 2 = 1$.
$\TMS^T_{1,2} = (2/3)/(1/3)\cdot x^T_2/x^T_1 = 2\cdot (4/3)/(8/3) = 2\cdot (1/2) = 1$.
$\TMS^H = \TMS^T = 1 = p_1^*/p_2^*.\ \checkmark$

\textit{Sanity econĂ´mico.} Apesar das dotaĂ§Ăľes simĂŠtricas $(\,3,\,1\,)$ vs.\ $(\,1,\,3\,)$, a alocaĂ§ĂŁo de equilĂ­brio ĂŠ \emph{simĂŠtrica espelhada} ao consumo desejado: $H$ acaba consumindo $(4/3, 8/3)$ -- mais capital -- e $T$ acaba consumindo $(8/3, 4/3)$ -- mais energia. Cada agente vendeu o que sobrava da dotaĂ§ĂŁo ($H$ vendeu 5/3 unidade de energia; $T$ vendeu 5/3 de capital) e o preĂ§o relativo ficou em $p_1^* = 1$ por simetria estrutural.
\end{solution}

\part[5] \textbf{1Âş Teorema do Bem-Estar em troca pura.}

\textbf{(i)} Enuncie o 1Âş TBE para economias de troca pura, identificando explicitamente a hipĂłtese sobre as preferĂŞncias e a hipĂłtese sobre o conceito de equilĂ­brio.

\textbf{(ii)} Prove o 1Âş TBE por contradiĂ§ĂŁo (3 passos: definiĂ§ĂŁo de Pareto-superioridade $\to$ orĂ§amento $\to$ Walras).

\textbf{(iii)} Verifique \emph{numericamente} que o equilĂ­brio computado em (a) ĂŠ Pareto-eficiente: mostre que $\TMS^H = \TMS^T$ no ponto $(x^{H*}, x^{T*})$, condiĂ§ĂŁo necessĂĄria e suficiente para Pareto-eficiĂŞncia interior em economia $2\times 2$ com preferĂŞncias CD.

\begin{solution}
\textbf{(i) Enunciado.} Em economia de troca pura $I \times L$ com preferĂŞncias $\succeq^i$ que satisfaĂ§am a propriedade de \emph{NĂŁo-Saciedade Local} (LNS) em todo $x^i \in \mathbb{R}^L_+$, todo equilĂ­brio competitivo $(p^*, x^*)$ ĂŠ Pareto-eficiente.

\textit{HipĂłtese de preferĂŞncia:} LNS -- para todo $x$ e toda vizinhanĂ§a aberta $V$ de $x$, existe $\tilde x \in V$ com $\tilde x \succ x$. (Mais geral que monotonicidade estrita; admite exemplos com saciedade direcional desde que sempre haja \emph{alguma} direĂ§ĂŁo de melhoria.)

\textit{HipĂłtese de equilĂ­brio:} todo agente ĂŠ tomador de preĂ§o e $x^{i*}$ resolve o UMP em $p^*$; $\sum_i x^{i*} = \sum_i \omega^i$ (viabilidade).

\textbf{(ii) Prova por contradiĂ§ĂŁo.} Suponha que existe alocaĂ§ĂŁo viĂĄvel $\tilde x = (\tilde x^1, \ldots, \tilde x^I)$ com $u^i(\tilde x^i) \geq u^i(x^{i*})$ para todo $i$ e estrita para algum $j$.

\textit{Passo 1.} Como $\tilde x^j$ ĂŠ estritamente preferida a $x^{j*}$ por $j$, e $x^{j*}$ resolve UMP em $p^*$, $\tilde x^j$ tem que estar fora do orĂ§amento de $j$ -- ou seja,
\[
    p^* \cdot \tilde x^j > p^* \cdot \omega^j.
\]

\textit{Passo 2 (LNS).} Para $i \neq j$ com $u^i(\tilde x^i) \geq u^i(x^{i*})$: \emph{afirmamos} $p^* \cdot \tilde x^i \geq p^* \cdot \omega^i$. Suponha por absurdo $p^* \cdot \tilde x^i < p^* \cdot \omega^i$ -- estritamente dentro do orĂ§amento. Por LNS, existe $\hat x^i$ em vizinhanĂ§a de $\tilde x^i$ com $\hat x^i \succ \tilde x^i \succeq x^{i*}$. Por continuidade do produto interno, $\hat x^i$ ainda ĂŠ factĂ­vel em $p^*$, contradizendo a hipĂłtese de que $x^{i*}$ resolve UMP. Logo $p^* \cdot \tilde x^i \geq p^* \cdot \omega^i$.

\textit{Passo 3.} Somando sobre todos os $i$ (com a desigualdade estrita em $j$):
\[
    \sum_i p^* \cdot \tilde x^i > \sum_i p^* \cdot \omega^i.
\]
Mas viabilidade exige $\sum_i \tilde x^i = \sum_i \omega^i$, logo $\sum_i p^* \cdot \tilde x^i = \sum_i p^* \cdot \omega^i$. ContradiĂ§ĂŁo. $\blacksquare$

\textbf{(iii) VerificaĂ§ĂŁo numĂŠrica em $x^*$ da parte (a).}
\[
    \TMS^H_{1,2}(x^{H*}) = \frac{(1/3)\, x^H_2}{(2/3)\, x^H_1} = \frac{1}{2}\cdot \frac{8/3}{4/3} = 1;\quad
    \TMS^T_{1,2}(x^{T*}) = \frac{(2/3)\, x^T_2}{(1/3)\, x^T_1} = 2\cdot \frac{4/3}{8/3} = 1.
\]
$\TMS^H = \TMS^T = 1$. TangĂŞncia das curvas de indiferenĂ§a -- $x^*$ estĂĄ na contract curve, logo ĂŠ Pareto-eficiente. $\checkmark$

\textit{ObservaĂ§ĂŁo.} No prĂłximo bloco (Parte II, Aula 5), repetiremos exatamente a prova do 1Âş TBE substituindo $\mathbb{R}^L$ por $\mathbb{R}^{LS}$ -- linha por linha, sem alteraĂ§ĂŁo estrutural. O argumento usa apenas LNS, UMP e viabilidade; nĂŁo usa nada da estrutura do espaĂ§o de bens.
\end{solution}

\part[5] \textbf{Tecnologia, PPF e $\TMT$.} Considere agora o \emph{lado da oferta} da economia. A energia firme do bem 1 ĂŠ produzida em dois setores -- hidrelĂŠtrico (com tecnologia $f_h$) e termelĂŠtrico (com tecnologia $f_t$) -- a partir de um fator agregado $L$ (mĂŁo-de-obra tĂŠcnica $+$ capital de operaĂ§ĂŁo, normalizado em unidades comparĂĄveis). As tecnologias sĂŁo
\[
    q_h = f_h(L_h) = 2\sqrt{L_h},\qquad q_t = f_t(L_t) = \sqrt{L_t},
\]
e a dotaĂ§ĂŁo total do fator ĂŠ $\bar L = L_h + L_t = 8$.

\textbf{(i)} Derive a equaĂ§ĂŁo implĂ­cita da PPF $\{(q_h, q_t) : q_h, q_t \geq 0\}$ no plano $(q_h, q_t)$ que esgota o fator. Demonstre que a PPF ĂŠ \emph{cĂ´ncava}.

\textbf{(ii)} Calcule $\TMT_{h, t} = -dq_t/dq_h\big|_{\text{PPF}}$ como funĂ§ĂŁo de $(q_h, q_t)$, por dois caminhos: (a) derivada implĂ­cita da PPF; (b) razĂŁo $\mathrm{PMa}_L^{h}/\mathrm{PMa}_L^{t}$ via regra da cadeia. Verifique que coincidem.

\textbf{(iii)} Esboce a PPF marcando os interceptos e o ponto $(q_h, q_t) = (4, 2)$ que serĂĄ o equilĂ­brio em (e).

\begin{solution}
\textbf{(i) PPF.} Inversa: $L_h = q_h^2/4$, $L_t = q_t^2$. RestriĂ§ĂŁo:
\[
    \frac{q_h^2}{4} + q_t^2 = 8,\quad q_h, q_t \geq 0. \quad \boxed{\phantom{x}}
\]
Em forma explĂ­cita: $q_t = \sqrt{8 - q_h^2/4}$, definida em $q_h \in [0, \sqrt{32}] = [0, 4\sqrt{2}]$.

\textbf{Concavidade.} A PPF ĂŠ cĂ´ncava no plano $(q_h, q_t)$ (com $q_t$ como funĂ§ĂŁo de $q_h$): $q_t'(q_h) = -q_h/(4 q_t)$ e
\[
    q_t''(q_h) = -\frac{(4 q_t) - q_h\cdot (4 q_t'(q_h))}{(4 q_t)^2}
    = -\frac{4 q_t + q_h^2/q_t}{16 q_t^2} = -\frac{4 q_t^2 + q_h^2}{16 q_t^3} < 0
\]
em todo o interior da PPF. Logo $q_t$ ĂŠ funĂ§ĂŁo estritamente cĂ´ncava de $q_h$ -- a PPF ĂŠ cĂ´ncava.

\textit{Por que cĂ´ncava?} Cada $f_\ell$ ĂŠ estritamente cĂ´ncava (raiz quadrada). Com fatores escalares $\sqrt{L}$, alocar tudo a um setor ĂŠ ineficiente -- a curvatura de cada $f_\ell$ ĂŠ o que produz a transformaĂ§ĂŁo cĂ´ncava no espaĂ§o de produtos.

\textbf{(ii) $\TMT$ por dois caminhos.}

\emph{Caminho A (derivada implĂ­cita):} de $q_h^2/4 + q_t^2 = 8$, diferenciando: $q_h\, dq_h/2 + 2 q_t\, dq_t = 0 \Rightarrow dq_t/dq_h = -q_h/(4 q_t)$, logo
\[
    \TMT_{h, t} = \frac{q_h}{4 q_t}.
\]

\emph{Caminho B (razĂŁo de PMa).} $\mathrm{PMa}_L^{h} = f_h'(L_h) = 1/\sqrt{L_h}$; $\mathrm{PMa}_L^{t} = f_t'(L_t) = 1/(2\sqrt{L_t})$. Substituindo $L_h = q_h^2/4 \Rightarrow \sqrt{L_h} = q_h/2$ e $L_t = q_t^2 \Rightarrow \sqrt{L_t} = q_t$:
\[
    \mathrm{PMa}_L^{h} = \frac{2}{q_h},\quad \mathrm{PMa}_L^{t} = \frac{1}{2 q_t},\quad
    \frac{\mathrm{PMa}_L^{h}}{\mathrm{PMa}_L^{t}} = \frac{2/q_h}{1/(2 q_t)} = \frac{4 q_t}{q_h}.
\]
\textit{AtenĂ§ĂŁo ao sentido:} $\TMT_{h, t}$ no caminho A ĂŠ $-dq_t/dq_h$ (quanto se desiste de $q_t$ por unidade extra de $q_h$); no caminho B, a razĂŁo $\mathrm{PMa}^h/\mathrm{PMa}^t$ ĂŠ a quantidade de $q_h$ que se ganha desviando uma unidade de $L$ de $t$ para $h$, dividida pela quantidade de $q_t$ perdida. As duas grandezas sĂŁo \emph{recĂ­procas}, e coincidem na convenĂ§ĂŁo que adotamos: $\TMT_{h, t} \equiv -dq_t/dq_h = q_h/(4q_t)$. O caminho B nesta convenĂ§ĂŁo dĂĄ $\mathrm{PMa}^t/\mathrm{PMa}^h = (1/(2q_t))/(2/q_h) = q_h/(4 q_t)$. $\checkmark$ \emph{Bater na ordem dos Ă­ndices}: $\TMT_{h,t}$ = ``$q_t$ perdido por $q_h$ ganho'' = $\mathrm{PMa}^t/\mathrm{PMa}^h$.

\textbf{(iii) EsboĂ§o da PPF.}

\part[4] \textbf{Robinson Crusoe descentralizado e equilĂ­brio competitivo de produĂ§ĂŁo.} O ``paĂ­s'' ĂŠ representado por um agente Robinson com preferĂŞncias sobre os dois bens: $u(q_h, q_t) = q_h^{1/2}\, q_t^{1/2}$ (Cobb-Douglas simĂŠtrica sobre energia hidro e tĂŠrmica -- estilizaĂ§ĂŁo: ambas energiam o mesmo grid e o consumidor mĂŠdio ĂŠ indiferente entre fontes, dada disponibilidade). A firma ĂŠ representativa, com tecnologia conjunta $\{(q_h, q_t) : q_h^2/4 + q_t^2 \leq 8\}$ -- a PPF do item (d). Robinson ĂŠ simultaneamente \emph{dono integral} da firma ($\theta = 1$) e Ăşnico consumidor.

\textbf{(i)} Enuncie o problema do planejador (max $u$ sujeito a estar na PPF) e derive a CPO interior $\TMS_{h,t} = \TMT_{h,t}$. Resolva o sistema e obtenha $(q_h^*, q_t^*)$.

\textbf{(ii)} Mostre que o mesmo $(q_h^*, q_t^*)$ ĂŠ o equilĂ­brio competitivo descentralizado: a firma escolhe $(q_h, q_t)$ maximizando lucro $\pi = p_h q_h + p_t q_t$ s.a.\ $q_h^2/4 + q_t^2 \leq 8$, e Robinson resolve UMP com renda $m = \pi^*$. Derive os preĂ§os $(p_h^*, p_t^*)$ que sustentam o equilĂ­brio (numerĂĄrio $p_h = 1$).

\textbf{(iii)} Verifique a igualdade canĂ´nica $\TMS = \TMT = p_h^*/p_t^*$ no ponto.

\begin{solution}
\textbf{(i) Planejador.} $\max_{q_h, q_t} q_h^{1/2}q_t^{1/2}$ s.a.\ $q_h^2/4 + q_t^2 = 8$. CPO interior: $\TMS_{h,t} = \TMT_{h,t}$.
\[
    \TMS_{h,t} = \frac{\partial u/\partial q_h}{\partial u/\partial q_t} = \frac{(1/2)(q_t/q_h)^{1/2}}{(1/2)(q_h/q_t)^{1/2}} = \frac{q_t}{q_h};
\]
\[
    \TMT_{h,t} = \frac{q_h}{4 q_t}.
\]
Igualando: $q_t/q_h = q_h/(4 q_t) \Rightarrow 4 q_t^2 = q_h^2 \Rightarrow q_h = 2 q_t$ (raiz positiva). Substituindo na PPF: $(2 q_t)^2/4 + q_t^2 = q_t^2 + q_t^2 = 2 q_t^2 = 8 \Rightarrow q_t^* = 2$, $q_h^* = 4$, $L_h = q_h^2/4 = 4$, $L_t = q_t^2 = 4$ (trabalho dividido em partes iguais). $\boxed{(q_h^*, q_t^*) = (4, 2)}$.

\textbf{(ii) DescentralizaĂ§ĂŁo.} \emph{Firma:} resolve $\max p_h q_h + p_t q_t$ s.a.\ $q_h^2/4 + q_t^2 \leq 8$. Lagrangiano: $\mathcal{L} = p_h q_h + p_t q_t - \mu (q_h^2/4 + q_t^2 - 8)$. CPOs:
\[
    p_h = \mu\, q_h/2,\qquad p_t = \mu\, 2 q_t \;\Longrightarrow\; \frac{p_h}{p_t} = \frac{q_h/2}{2 q_t} = \frac{q_h}{4 q_t} = \TMT_{h,t}.\ \checkmark
\]

Em $(q_h^*, q_t^*) = (4, 2)$: $p_h/p_t = 4/8 = 1/2$. Com $p_h = 1$ numerĂĄrio, $\boxed{p_t^* = 2}$. Lucro:
\[
    \pi^* = p_h q_h^* + p_t q_t^* = 1\cdot 4 + 2\cdot 2 = 8.
\]

\emph{Robinson como consumidor.} Renda $m = \pi^* = 8$. UMP CD simĂŠtrica: $q_h = (1/2)\cdot m/p_h = 4$; $q_t = (1/2)\cdot m/p_t = 2$. $\checkmark$ Coincide com a oferta da firma.

\textbf{(iii) Igualdade canĂ´nica.} $\TMS_{h,t}(q^*) = q_t^*/q_h^* = 2/4 = 1/2$. $\TMT_{h,t}(q^*) = q_h^*/(4 q_t^*) = 4/8 = 1/2$. $p_h^*/p_t^* = 1/2$. As trĂŞs grandezas coincidem:
\[
    \boxed{\;\TMS_{h,t} \;=\; \TMT_{h,t} \;=\; \frac{p_h^*}{p_t^*} \;=\; \frac{1}{2}.\;}
\]
\textbf{Mercado descentralizado replica a soluĂ§ĂŁo do planejador.} Esta ĂŠ a \emph{assinatura matemĂĄtica} do 1Âş TBE com produĂ§ĂŁo, no caso $I = J = 1$.

\textit{Sanity econĂ´mico.} Com tecnologias hidro mais produtivas ($f_h = 2\sqrt{L_h}$ vs.\ $f_t = \sqrt{L_t}$) e preferĂŞncias simĂŠtricas, o sistema produz \emph{mais} energia hidro ($q_h^* = 4$) que tĂŠrmica ($q_t^* = 2$), e o preĂ§o relativo da tĂŠrmica fica \emph{mais alto} ($p_t^* = 2$, $p_h^* = 1$) -- a tĂŠrmica ĂŠ o ``recurso mais escasso por unidade de fator''. Em proporĂ§Ăľes aproximadas, hidro responde por $4/(4+2) = 67\%$ do mix em quantidade fĂ­sica e por $4/(4+4) = 50\%$ em valor -- estilizaĂ§ĂŁo que ecoa, qualitativamente, a participaĂ§ĂŁo da hidro na matriz brasileira (\emph{Boletim Mensal de OperaĂ§ĂŁo}, ONS, 2023: $\sim$60\% fĂ­sica hidro). Em Q2, esse preĂ§o relativo serĂĄ deslocado pela introduĂ§ĂŁo de risco hidrolĂłgico.
\end{solution}

\part[5] \textbf{EficiĂŞncia produtiva, eficiĂŞncia alocativa e o 1Âş TBE com produĂ§ĂŁo.}

\textbf{(i)} Defina os trĂŞs conceitos de eficiĂŞncia relevantes em economia com produĂ§ĂŁo: (a) eficiĂŞncia produtiva (alocaĂ§ĂŁo eficiente do fator entre setores: $\mathrm{PMa}_L^{h}/\mathrm{PMa}_L^{t} = $ const.\ ao longo da PPF); (b) eficiĂŞncia alocativa em consumo (tangĂŞncia das ICs no agregado); (c) eficiĂŞncia em \emph{produto-mix} ($\TMS = \TMT$).

\textbf{(ii)} Mostre que o equilĂ­brio competitivo do item (e) satisfaz \emph{simultaneamente} as trĂŞs -- e \emph{argumente} que essa simultaneidade nĂŁo ĂŠ coincidĂŞncia: ĂŠ o conteĂşdo do 1Âş TBE estendido a produĂ§ĂŁo.

\textbf{(iii)} \textit{Pergunta conceitual.} Suponha uma reforma regulatĂłria que, por algum motivo, introduza uma \textbf{cunha} entre $\TMS$ e $\TMT$ -- por exemplo, uma alĂ­quota tributĂĄria assimĂŠtrica que faĂ§a $p_h^{\text{firma}} \neq p_h^{\text{consumidor}}$. Argumente, em uma frase para cada, qual das trĂŞs eficiĂŞncias do (i) ĂŠ genericamente preservada e qual(is) ĂŠ(sĂŁo) destruĂ­da(s).

\begin{solution}
\textbf{(i) TrĂŞs conceitos de eficiĂŞncia (com produĂ§ĂŁo).}

\emph{Produtiva.} O fator $L$ estĂĄ alocado entre os setores de modo que nĂŁo hĂĄ realocaĂ§ĂŁo capaz de produzir mais de algum bem sem reduzir o outro -- equivalente a $L$ esgotado e razĂŁo $\mathrm{PMa}_L^{h}/\mathrm{PMa}_L^{t}$ comum a todos os setores que utilizam $L$ (igualaĂ§ĂŁo dos valores marginais).

\emph{Alocativa em consumo.} A alocaĂ§ĂŁo do produto entre os agentes ĂŠ Pareto-eficiente -- equivalente, em economia com agentes interiores, Ă  tangĂŞncia $\TMS^A = \TMS^B$ (Bloco 1.c, durante-aula).

\emph{EficiĂŞncia de produto-mix.} O \emph{vetor produzido} $(q_h, q_t)$ estĂĄ na PPF \emph{e} a inclinaĂ§ĂŁo local da PPF coincide com a inclinaĂ§ĂŁo local das ICs agregadas: $\TMS = \TMT$. Sem essa condiĂ§ĂŁo, a economia produz a quantidade ``errada'' de cada bem dada a preferĂŞncia social.

\textbf{(ii) VerificaĂ§ĂŁo simultĂ˘nea no equilĂ­brio do item (e).}

\emph{Produtiva:} dado o numerĂĄrio $p_h = 1$, $p_t = 2$, $w$ ĂŠ determinado pela CPO de fator -- $w = p_h\cdot \mathrm{PMa}^h_L = 1\cdot (2/q_h^*) = 1/2$, e tambĂŠm $w = p_t\cdot \mathrm{PMa}^t_L = 2\cdot (1/(2 q_t^*)) = 1/2$. $\checkmark$ ($L$ tem o mesmo preĂ§o de sombra nos dois setores -- o salĂĄrio $w = 1/2$ -- garantindo eficiĂŞncia produtiva.)

\emph{Alocativa em consumo:} $I = 1$, trivialmente Pareto-eficiente (o agente Ăşnico maximiza sua utilidade sujeita ao orĂ§amento agregado).

\emph{Produto-mix:} $\TMS_{h,t} = \TMT_{h,t} = 1/2$ (item e-iii). $\checkmark$

\emph{NĂŁo-coincidĂŞncia.} O 1Âş TBE estendido a produĂ§ĂŁo diz: sob LNS $+$ maximizaĂ§ĂŁo de lucro de cada firma, o EC ĂŠ Pareto-eficiente -- o que, em economia com produĂ§ĂŁo, significa as \emph{trĂŞs} simultĂ˘neamente. A prova ganha um quarto passo em relaĂ§ĂŁo Ă  versĂŁo de troca pura: para cada firma $j$ e para cada $\tilde y^j \in Y^j$, $p^*\cdot \tilde y^j \leq \pi^{*j} = p^*\cdot y^{*j}$ \emph{por definiĂ§ĂŁo} de $y^{*j}$ como maximizador. Substituindo na soma da Aula 4 (Bloco 3.b), a contradiĂ§ĂŁo segue. As trĂŞs eficiĂŞncias saem juntas porque o Ăşnico preĂ§o $p^*$ alinha simultaneamente decisĂľes da firma, do consumidor e do mercado de fator.

\textbf{(iii) Cunha entre $\TMS$ e $\TMT$.}

A eficiĂŞncia \emph{produtiva} ĂŠ \textbf{preservada} -- desde que a cunha incida sobre o \emph{produto} (nĂŁo sobre o fator), a firma continua minimizando custo por unidade produzida, igualando a razĂŁo de PMa de fator entre setores via $w$.

A eficiĂŞncia \emph{alocativa em consumo} tambĂŠm ĂŠ preservada com $I = 1$ trivialmente; com $I > 1$, ĂŠ preservada se a cunha for \emph{uniforme} entre agentes (cada consumidor enfrenta o mesmo $p^{\text{cons}}$).

A eficiĂŞncia de \emph{produto-mix} ĂŠ \textbf{destruĂ­da}: $\TMS$ ajusta-se ao $p^{\text{cons}}_h/p^{\text{cons}}_t$ enquanto $\TMT$ ajusta-se ao $p^{\text{firma}}_h/p^{\text{firma}}_t$. Quando o regulador insere a cunha tributĂĄria, abre-se espaĂ§o entre $\TMS$ e $\TMT$ -- a economia produz quantidades ``erradas'' relativamente ao desejo agregado dos consumidores. Esta ĂŠ a estrutura bĂĄsica do peso-morto de tributaĂ§ĂŁo sobre produto, e reaparecerĂĄ em Q4 ao discutirmos o efeito da \emph{repactuaĂ§ĂŁo do GSF} sobre os incentivos de despacho.
\end{solution}

\question \textbf{EquilĂ­brio Arrow-Debreu, seguro completo e SDF no setor elĂŠtrico} (30 pontos).

\begin{parts}
\part[5] \textbf{Bens contingentes e UMP em $\mathbb{R}^{LS}$.}

\textbf{(i)} Quantos bens distintos existem nesta economia? Escreva o vetor de consumo $x^H \in \mathbb{R}^{LS}$ e o vetor de preĂ§os $p \in \mathbb{R}^{LS}$.

\textbf{(ii)} Escreva a funĂ§ĂŁo utilidade EU $u^H(x^H)$ e $u^T(x^T)$ explicitamente.

\textbf{(iii)} Escreva o problema UMP de cada agente em $\mathbb{R}^{LS}$.

\begin{solution}
\textbf{(i)} $L = 1$, $|S| = 2$, logo $LS = 2$ \emph{bens contingentes distintos}: ``energia se favorĂĄvel'' ($\ell, f$) e ``energia se seca'' ($\ell, s$). Como hĂĄ um Ăşnico bem fĂ­sico ($L = 1$), suprimimos o Ă­ndice $\ell$ e escrevemos $x^i = (x^i_f, x^i_s) \in \mathbb{R}^2_+$. Vetor de preĂ§os contingentes $p = (p_f, p_s) \in \mathbb{R}^2_+$ -- preĂ§o, em $t = 0$, de uma promessa de entrega de uma unidade fĂ­sica de energia condicional ao estado realizado.

\textbf{(ii) EU:}
\[
    u^H(x^H) = \pi_f\, \ln x^H_f + \pi_s\, \ln x^H_s = \tfrac{1}{2}\ln x^H_f + \tfrac{1}{2}\ln x^H_s.
\]
\[
    u^T(x^T) = \pi_f\, x^T_f + \pi_s\, x^T_s = \tfrac{1}{2}\, x^T_f + \tfrac{1}{2}\, x^T_s = \tfrac{1}{2}(x^T_f + x^T_s).
\]
$T$ tem utilidade EU \emph{linear} -- captura neutralidade ao risco.

\textbf{(iii) UMP em $\mathbb{R}^{LS}$.} Cada agente $i$ resolve
\[
    \max_{x^i \geq 0}\, u^i(x^i)\quad \text{s.a.}\quad p_f\, x^i_f + p_s\, x^i_s \leq p_f\, \omega^i_f + p_s\, \omega^i_s.
\]
\textit{ObservaĂ§ĂŁo importante.} A restriĂ§ĂŁo ĂŠ \emph{uma sĂł} -- a Ăşnica ``cesta'' enfrentada pelo agente ĂŠ a transferĂŞncia de valor entre estados via mercado AD em $t = 0$. Toda a estrutura algĂŠbrica do UMP da Aula 1 sobrevive; o que muda ĂŠ a \emph{interpretaĂ§ĂŁo dos Ă­ndices}. Esta ĂŠ a observaĂ§ĂŁo central da Aula 5.
\end{solution}

\part[6] \textbf{EquilĂ­brio AD: condiĂ§ĂŁo imposta pelo agente neutro e cĂĄlculo do preĂ§o relativo.}

\textbf{(i)} Resolva o UMP de $T$. Como $u^T$ ĂŠ linear, mostre que $T$ \emph{exige} (em Ăłtimo interior) $p^*_f/p^*_s = \pi_f/\pi_s$ -- preĂ§os \textbf{atuarialmente justos}. Caso contrĂĄrio, $T$ deseja consumo infinito no estado mais barato relativamente Ă  sua probabilidade.

\textbf{(ii)} Sob a normalizaĂ§ĂŁo canĂ´nica $p^*_f + p^*_s = 1$, obtenha $(p^*_f, p^*_s)$ explicitamente.

\textbf{(iii)} \emph{ComentĂĄrio econĂ´mico.} Explique, em uma frase, o sentido de o agente neutro ``determinar'' os preĂ§os de equilĂ­brio: por que ele tem essa influĂŞncia desproporcional?

\begin{solution}
\textbf{(i) UMP de $T$.} $T$ resolve $\max\, (1/2)(x^T_f + x^T_s)$ s.a.\ $p_f x^T_f + p_s x^T_s \leq p_f\, 12 + p_s\, 24$. A funĂ§ĂŁo objetivo ĂŠ linear com coeficientes $(\pi_f, \pi_s) = (1/2, 1/2)$.

\emph{Caso $p_f/\pi_f < p_s/\pi_s$}: $T$ obtĂŠm mais utilidade por R\$ comprando o bem $f$ -- desejaria $x^T_f \to \infty$, $x^T_s = 0$. InsustentĂĄvel em equilĂ­brio com viabilidade $x^T_f + x^H_f = 36$.

\emph{Caso $p_f/\pi_f > p_s/\pi_s$}: simĂŠtrico, $x^T_s \to \infty$.

\emph{Caso $p_f/\pi_f = p_s/\pi_s$}: $T$ ĂŠ indiferente ao longo da reta orĂ§amentĂĄria -- aceita qualquer alocaĂ§ĂŁo compatĂ­vel com viabilidade. Esta ĂŠ a Ăşnica configuraĂ§ĂŁo compatĂ­vel com EC interior.

Logo $\boxed{p^*_f/p^*_s = \pi_f/\pi_s = 1.}$ Equivalentemente, $p^*_f = p^*_s$ -- preĂ§os \textbf{atuarialmente justos}.

\textbf{(ii) NormalizaĂ§ĂŁo.} $p^*_f + p^*_s = 1$ + $p^*_f = p^*_s$ $\Rightarrow$ $\boxed{p^*_f = p^*_s = 0{,}5.}$

\textbf{(iii) ComentĂĄrio.} O agente neutro absorve qualquer desvio dos preĂ§os relativos da relaĂ§ĂŁo atuarial -- ĂŠ ele quem ``carrega o risco'' marginal sem cobrar prĂŞmio -- e por isso \emph{ancora} o preĂ§o relativo no valor sem prĂŞmio de risco. Em equilĂ­brio com qualquer agente avesso $+$ agente neutro $+$ agregado livre de risco, $p^*_s/p^*_{s'} = \pi_s/\pi_{s'}$ \emph{sempre} -- a presenĂ§a do neutro elimina o prĂŞmio de risco dos preĂ§os-Arrow.
\end{solution}

\part[6] \textbf{CPO do agente avesso e seguro completo.}

\textbf{(i)} Derive a CPO interior do UMP de $H$ via Lagrangiano.

\textbf{(ii)} Combine a CPO com $p^*_f/p^*_s = 1$ obtido em (b) e a forma $v_H = \ln$ para mostrar que $x^{H*}_f = x^{H*}_s$ -- o gerador hidro fica \emph{totalmente segurado} no equilĂ­brio AD.

\textbf{(iii)} Calcule explicitamente $x^{H*}_f = x^{H*}_s = \bar c^H$ usando o orĂ§amento, e $x^{T*}$ usando viabilidade. Verifique o resultado contra a equaĂ§ĂŁo fundamental do SDF.

\begin{solution}
\textbf{(i) Lagrangiano de $H$.} $\mathcal{L} = (1/2)\ln x^H_f + (1/2)\ln x^H_s - \lambda(p_f x^H_f + p_s x^H_s - p_f\cdot 24 - p_s\cdot 12)$. CPOs:
\[
    \frac{\pi_f}{x^H_f} = \lambda\, p_f,\qquad \frac{\pi_s}{x^H_s} = \lambda\, p_s.
\]
RazĂŁo:
\[
    \boxed{\;\frac{\pi_f\, v_H'(x^H_f)}{\pi_s\, v_H'(x^H_s)} \;=\; \frac{\pi_f / x^H_f}{\pi_s / x^H_s} \;=\; \frac{p_f}{p_s}.\;}
\]
Esta ĂŠ a equaĂ§ĂŁo fundamental do SDF (com $v = \ln$, $v' = 1/x$).

\textbf{(ii) Seguro completo.} Substituindo $p^*_f/p^*_s = 1$ e $\pi_f = \pi_s = 1/2$:
\[
    \frac{(1/2)/x^H_f}{(1/2)/x^H_s} = 1 \;\Longleftrightarrow\; \frac{x^H_s}{x^H_f} = 1 \;\Longleftrightarrow\; x^{H*}_f = x^{H*}_s.
\]
$H$ ĂŠ \textbf{totalmente segurado} -- consumo independente do estado. \emph{AtenĂ§ĂŁo Ă  direĂ§ĂŁo do raciocĂ­nio:} a igualdade $\pi_f = \pi_s$ \emph{e} $p^*_f = p^*_s$ \emph{e} $v_H'$ injetora (ln estritamente cĂ´ncava) implicam $x^{H*}_f = x^{H*}_s$. Tirar qualquer das trĂŞs -- e.g., crenĂ§as heterogĂŞneas, ou seca menos provĂĄvel, ou Bernoulli linear -- quebra o seguro completo.

\textbf{(iii) CĂĄlculo numĂŠrico.} OrĂ§amento de $H$ em $p^* = (0{,}5;\,0{,}5)$:
\[
    0{,}5\cdot x^H_f + 0{,}5\cdot x^H_s = 0{,}5\cdot 24 + 0{,}5\cdot 12 = 18.
\]
Com $x^H_f = x^H_s = \bar c^H$: $\bar c^H = 18$. $\boxed{x^{H*} = (18, 18).}$

Por viabilidade: $x^{T*} = \bar\omega - x^{H*} = (36 - 18, 36 - 18) = \boxed{(18, 18)}.$

\emph{VerificaĂ§ĂŁo SDF.} Em $x^{H*} = (18, 18)$, $v_H'(18)/v_H'(18) = 1$, e $\pi_f/\pi_s = 1$, logo a equaĂ§ĂŁo fundamental dĂĄ $p_f/p_s = 1$. $\checkmark$ Em $x^{T*} = (18, 18)$, $v_T'(18)/v_T'(18) = 1$ trivialmente. $\checkmark$ A equaĂ§ĂŁo fundamental do SDF vale para \emph{ambos} os agentes simultaneamente -- como tem que valer em equilĂ­brio AD interior com mercado completo.

\textit{Sanity econĂ´mico setor elĂŠtrico.} O gerador hidro $H$, em autarquia, consumiria $(24, 12)$ -- volĂĄtil, com risco hidrolĂłgico ``cru''. ApĂłs o seguro completo via mercado AD, consome $(18, 18)$ -- estĂĄvel, sem risco. O agente neutro $T$ (tĂŠrmica/consumidor) absorve a volatilidade: em autarquia consumiria $(12, 24)$; apĂłs o equilĂ­brio, consome $(18, 18)$. \emph{AtenĂ§ĂŁo:} aqui $T$ tambĂŠm converge a $(18, 18)$ porque, sob preĂ§os atuarialmente justos, consumir o valor esperado da dotaĂ§ĂŁo maximiza utilidade linear sem custo extra (qualquer vetor compatĂ­vel com viabilidade dĂĄ a $T$ a mesma EU). Esta ĂŠ a estilizaĂ§ĂŁo AD canĂ´nica do que o \textbf{MRE} (Mecanismo de RealocaĂ§ĂŁo de Energia) tenta fazer entre os geradores hidro: mutualizar o risco hidrolĂłgico de modo que cada gerador receba sua \emph{Garantia FĂ­sica} ajustada pelo $\GSF$, abstraindo a alocaĂ§ĂŁo de tempo (favorĂĄvel vs.\ seca) em uma alocaĂ§ĂŁo de longo prazo equivalente ao seguro mĂştuo.
\end{solution}

\part[4] \textbf{GeneralizaĂ§ĂŁo do SDF e identificaĂ§ĂŁo do prĂŞmio de risco.}

A equaĂ§ĂŁo fundamental do SDF
\[
    \frac{p^*_s}{p^*_{s'}} \;=\; \frac{\pi^i_s\, (v^i)'(x^{i*}_s)}{\pi^i_{s'}\, (v^i)'(x^{i*}_{s'})}\quad \forall i
\]
admite a leitura $p^*_s = \pi_s\cdot m_s$ onde $m_s \equiv (v')(x^*_s)/[\sum_{s''} \pi_{s''}\, (v')(x^*_{s''})] \cdot$ (constante de normalizaĂ§ĂŁo) ĂŠ o \textbf{Stochastic Discount Factor}. Em mercado AD com agregado livre de risco (como o nosso), o SDF ĂŠ constante entre estados (como obtido em (b)).

\textbf{(i)} Mostre que, sob agregado livre de risco ($\bar\omega_f = \bar\omega_s$) e Bernoullis idĂŞnticas com risco partilhĂĄvel, o SDF ĂŠ constante e o prĂŞmio de risco entre estados ĂŠ zero.

\textbf{(ii)} \emph{CenĂĄrio alternativo (nĂŁo calcule alocaĂ§ĂŁo completa, sĂł o sinal qualitativo).} Suponha que o agregado deixe de ser livre de risco -- por exemplo, $\bar\omega = (40, 32)$ (mais energia total disponĂ­vel em $f$ que em $s$, refletindo que mesmo a tĂŠrmica nĂŁo compensa toda a falta hidro em estados de seca). Argumente: (a) o SDF se torna constante entre estados ou variĂĄvel? (b) o estado-seca ganha um \emph{prĂŞmio} ou \emph{desconto} no preĂ§o relativo?

\begin{solution}
\textbf{(i) Agregado livre de risco $\Rightarrow$ SDF constante.} Sob agregado $\bar\omega_f = \bar\omega_s = \bar\omega$ e Bernoullis idĂŞnticas avessas ($v$ comum, $v$ estritamente cĂ´ncava), a alocaĂ§ĂŁo Pareto-eficiente $x^{i*}$ tem que satisfazer $\TMS$ comum entre estados. Sob crenĂ§as de consenso $\pi_f = \pi_s$, a Ăşnica soluĂ§ĂŁo compatĂ­vel com $v$ injetora ĂŠ $x^{i*}_f = x^{i*}_s$ \emph{para cada} $i$. Logo $v'(x^{i*}_f)/v'(x^{i*}_s) = 1$ para cada $i$, e por isso a equaĂ§ĂŁo SDF entrega
\[
    \frac{p^*_f}{p^*_s} = \frac{\pi_f}{\pi_s}\cdot 1 = 1.
\]
SDF constante $m_f = m_s$. PrĂŞmio de risco entre estados: zero.

\textit{No nosso setup} ($H$ avessa ao risco, $T$ neutro), o resultado ĂŠ o mesmo \emph{porque} $T$ ĂŠ o \emph{marginal pricing agent}: como $T$ ĂŠ neutro, sua CPO forĂ§a $p^*_f/p^*_s = \pi_f/\pi_s$ independentemente do que faz $H$. VerificaĂ§ĂŁo: $T$ aceita qualquer alocaĂ§ĂŁo compatĂ­vel com viabilidade ao preĂ§o atuarial justo.

\textbf{(ii) Agregado com risco $\bar\omega = (40, 32)$.}

(a) \emph{SDF se torna variĂĄvel.} Em qualquer alocaĂ§ĂŁo Pareto-eficiente com agregado volĂĄtil e agentes \emph{coletivamente} avessos (ainda que apenas um seja avesso), o consumo agregado tem que cair em $s$ -- pelo menos um agente vĂŞ $x^*_s < x^*_f$, logo $v'(x^*_s) > v'(x^*_f)$ (concavidade), e a equaĂ§ĂŁo SDF entrega $p^*_s/p^*_f > \pi_s/\pi_f$, i.e., $m_s > m_f$. \textbf{O SDF cresce em estados ``ruins''} -- intuiĂ§ĂŁo central de Hansen-Jagannathan (1991) e do \emph{equity premium puzzle} (Mehra-Prescott, 1985).

(b) \emph{O estado-seca ganha um prĂŞmio.} O preĂ§o-Arrow $p^*_s$ por unidade de probabilidade ($p^*_s/\pi_s$) sobe acima do nĂ­vel atuarial justo. Em linguagem financeira: ativos que pagam mais em $s$ (por exemplo, contratos de \emph{hedge} contra seca como ``puts sobre PLD'') valem desproporcionalmente mais -- carregam ``prĂŞmio de seguro'' positivo. Esta ĂŠ a fundaĂ§ĂŁo microeconĂ´mica para a \textbf{precificaĂ§ĂŁo de risco hidrolĂłgico em mercados financeiros derivados} sobre PLD/preĂ§o-Arrow no contexto pĂłs-Lei 14.052/2020.

\emph{Em uma frase:} ``preĂ§os-Arrow sĂŁo probabilidades torcidas pela aversĂŁo ao risco -- a torĂ§ĂŁo ĂŠ zero quando o agregado ĂŠ estĂĄvel, e cresce com a volatilidade do agregado e com a curvatura de $v$.''
\end{solution}

\end{solution}

\begin{center}\large\textbf{Parte III --- Arrow-Debreu II: Radner sequencial, mercados (in)completos e equaĂ§ĂŁo de Euler} \quad (35 pontos)\end{center}

\textit{ConexĂŁo com a Aula 6.} Os itens abaixo trabalham: (i) implementaĂ§ĂŁo do equilĂ­brio de Q2 via Radner sequencial com 2 ativos (CCEAR-bond + aĂ§ĂŁo contingente hidro); (ii) cĂĄlculo de portfolio replicador para uma aĂ§ĂŁo de Arrow; (iii) ampliaĂ§ĂŁo para 3 estados com manutenĂ§ĂŁo de 2 ativos $\Rightarrow$ \textbf{mercado incompleto}; (iv) diagnĂłstico da Pareto-ineficiĂŞncia genĂŠrica (Hart 1975) e do efeito ``mais ativos pode piorar'' (Geanakoplos-Polemarchakis, 1986); (v) especializaĂ§ĂŁo do SDF para a equaĂ§ĂŁo de Euler intertemporal e ligaĂ§ĂŁo com a SELIC.

\noindent\textbf{Setup geral de Q3.} Mantemos a economia de Q2 ($I = 2$, $L = 1$, $S = \{f, s\}$, $\pi = (1/2, 1/2)$, $v_H = \ln$, $v_T = c$, $\omega^H = (24, 12)$, $\omega^T = (12, 24)$). \question \textbf{Radner sequencial, mercados incompletos e Euler intertemporal} (35 pontos).

\begin{parts}
\part[5] \textbf{Estrutura de ativos e completude do mercado.}

\textbf{(i)} Calcule $\det A$ e $\rank A$. O mercado ĂŠ completo?

\textbf{(ii)} Aplique a relaĂ§ĂŁo de \textbf{nĂŁo-arbitragem} $q^*_j = \sum_s p^*_s\, A_{sj}$ usando os preĂ§os-Arrow $p^* = (0{,}5;\,0{,}5)$ obtidos em Q2(b) para calcular os preĂ§os de equilĂ­brio dos dois ativos: $q^*_{\text{bond}}$ e $q^*_{\text{hidro}}$.

\textbf{(iii)} Comente: por que $q^*_{\text{hidro}} = 0{,}9 < 1 = E[D^{\text{hidro}}]$? \emph{Pista:} olhe para a relaĂ§ĂŁo entre o preĂ§o atuarial-justo e o que o ativo hidro entrega em cada estado.

\begin{solution}
\textbf{(i)} $\det A = 1\cdot 0{,}6 - 1\cdot 1{,}2 = -0{,}6 \neq 0$, logo $\rank A = 2 = |S|$. \textbf{Mercado completo.} Os dois ativos \emph{spannem} $\mathbb{R}^{|S|} = \mathbb{R}^2$ -- toda aĂ§ĂŁo Arrow ĂŠ replicĂĄvel.

\textbf{(ii) NĂŁo-arbitragem.}
\[
    q^*_{\text{bond}} = p^*_f\cdot 1 + p^*_s\cdot 1 = 0{,}5 + 0{,}5 = 1.
\]
\[
    q^*_{\text{hidro}} = p^*_f\cdot 1{,}2 + p^*_s\cdot 0{,}6 = 0{,}6 + 0{,}3 = 0{,}9.
\]
$\boxed{q^* = (q^*_{\text{bond}}, q^*_{\text{hidro}}) = (1{,}00;\,0{,}90).}$

\textbf{(iii) Por que $q^*_{\text{hidro}} = 0{,}9$ e nĂŁo $1$?} \emph{NĂŁo} ĂŠ por aversĂŁo ao risco -- nesta economia, sob agregado livre de risco e agente neutro presente, os preĂ§os-Arrow sĂŁo exatamente atuarial-justos ($p^*_f = p^*_s = 1/2$, sem prĂŞmio de risco), e o SDF ĂŠ constante. Logo o preĂ§o da aĂ§ĂŁo hidro ĂŠ simplesmente seu \emph{valor esperado descontado a SDF unitĂĄrio}:
\[
    q^*_{\text{hidro}} = E[m\cdot D^{\text{hidro}}] = m\cdot E[D^{\text{hidro}}] = 1\cdot \tfrac{1{,}2 + 0{,}6}{2} = 0{,}9.
\]
A diferenĂ§a $q^*_{\text{hidro}} = 0{,}9 < 1 = q^*_{\text{bond}}$ reflete que o ativo hidro \emph{paga em mĂŠdia menos} que o bond por contrato (mĂŠdia $0{,}9$ MWh vs.\ $1$ MWh), e \emph{nĂŁo} um prĂŞmio de risco. Em economia com agregado volĂĄtil e sem agente neutro (Q2(e)), $q^*_{\text{hidro}}$ seria \emph{ainda mais baixo} relativamente a $E[D^{\text{hidro}}]$ -- o ativo hidro paga \emph{relativamente menos no estado seca} (estado de stress), e em uma economia com aversĂŁo ao risco isso reduz seu preĂ§o (prĂŞmio de risco positivo, na linguagem de Lucas 1978).
\end{solution}

\part[7] \textbf{Portfolio replicador e equivalĂŞncia AD$\leftrightarrow$Radner.}

\textbf{(i)} Calcule o portfolio $\theta^{(f)} = (\theta^{(f)}_{\text{bond}}, \theta^{(f)}_{\text{hidro}})$ que replica a \textbf{aĂ§ĂŁo Arrow} para o estado $f$, ou seja, $A\theta^{(f)} = e_f = (1, 0)$.

\textbf{(ii)} Calcule o portfolio $\theta^{(s)}$ que replica $e_s = (0, 1)$.

\textbf{(iii)} Verifique \textbf{nĂŁo-arbitragem em ambos}: $q^* \cdot \theta^{(f)} = p^*_f$ e $q^* \cdot \theta^{(s)} = p^*_s$.

\textbf{(iv)} Calcule o portfolio $\theta^{H*}$ que implementa, em mercado de Radner, o consumo de equilĂ­brio $x^{H*} = (18, 18)$ obtido em Q2(c). Verifique a restriĂ§ĂŁo em $t = 0$: $q^* \cdot \theta^{H*} = 0$.

\textbf{(v)} Conclua: AD canĂ´nico (Q2) e Radner com 2 ativos sob $\rank A = 2$ entregam a \emph{mesma} alocaĂ§ĂŁo. Esta ĂŠ a equivalĂŞncia de Arrow (1953) / Radner (1972).

\begin{solution}
\textbf{(i) Replicar $e_f = (1, 0)$.} Sistema $A\theta = e_f$:
\[
    \begin{cases}
    \theta_{\text{bond}} + 1{,}2\, \theta_{\text{hidro}} = 1 \\
    \theta_{\text{bond}} + 0{,}6\, \theta_{\text{hidro}} = 0
    \end{cases}
\]
Subtraindo: $0{,}6\, \theta_{\text{hidro}} = 1 \Rightarrow \theta_{\text{hidro}} = 5/3$. Da segunda: $\theta_{\text{bond}} = -0{,}6\cdot 5/3 = -1$. Logo
\[
    \boxed{\theta^{(f)} = (-1,\, 5/3).}
\]
\emph{InterpretaĂ§ĂŁo econĂ´mica:} para receber 1 MWh em $f$ e nada em $s$, o agente \emph{vende a descoberto} 1 unidade de bond (recebe R\$ 1 em $t = 0$) e \emph{compra} $5/3$ unidades da aĂ§ĂŁo hidro (paga $5/3 \cdot 0{,}9 = 3/2$ R\$). Net em $t = 0$: $-1 + 3/2 = 0{,}5 = p^*_f$. $\checkmark$

\textbf{(ii) Replicar $e_s = (0, 1)$.}
\[
    \begin{cases}
    \theta_{\text{bond}} + 1{,}2\, \theta_{\text{hidro}} = 0 \\
    \theta_{\text{bond}} + 0{,}6\, \theta_{\text{hidro}} = 1
    \end{cases}
\]
Subtraindo: $0{,}6\, \theta_{\text{hidro}} = -1 \Rightarrow \theta_{\text{hidro}} = -5/3$. Da segunda: $\theta_{\text{bond}} = 1 + 0{,}6\cdot 5/3 = 1 + 1 = 2$. Logo
\[
    \boxed{\theta^{(s)} = (2,\, -5/3).}
\]

\textbf{(iii) VerificaĂ§ĂŁo de nĂŁo-arbitragem.} $q^* = (1; 0{,}9)$.
\[
    q^* \cdot \theta^{(f)} = 1\cdot (-1) + 0{,}9\cdot (5/3) = -1 + 1{,}5 = 0{,}5 = p^*_f. \quad \checkmark
\]
\[
    q^* \cdot \theta^{(s)} = 1\cdot 2 + 0{,}9\cdot (-5/3) = 2 - 1{,}5 = 0{,}5 = p^*_s. \quad \checkmark
\]
A nĂŁo-arbitragem $q^* \cdot \theta^{(s)} = p^*_s$ ĂŠ exatamente a definiĂ§ĂŁo: o custo em $t = 0$ de comprar uma unidade de promessa contingente ao estado $s$ via portfolio ĂŠ o preĂ§o-Arrow desse estado.

\textbf{(iv) Portfolio que implementa $x^{H*} = (18, 18)$.} Em Radner, $x^{H} = \omega^H + A\theta^H$, logo $A\theta^{H*} = x^{H*} - \omega^H = (18 - 24,\, 18 - 12) = (-6,\, 6)$.

Resolvendo $A\theta^{H*} = (-6, 6)$:
\[
    \begin{cases}
    \theta^H_{\text{bond}} + 1{,}2\, \theta^H_{\text{hidro}} = -6 \\
    \theta^H_{\text{bond}} + 0{,}6\, \theta^H_{\text{hidro}} = 6
    \end{cases}
\]
Subtraindo: $0{,}6\, \theta^H_{\text{hidro}} = -12 \Rightarrow \theta^H_{\text{hidro}} = -20$. Da segunda: $\theta^H_{\text{bond}} = 6 + 0{,}6\cdot 20 = 6 + 12 = 18$.
\[
    \boxed{\theta^{H*} = (18,\, -20).}
\]
\emph{InterpretaĂ§ĂŁo:} $H$ \emph{compra} 18 contratos CCEAR-bond (receberĂĄ 18 MWh garantidos em qualquer estado) e \emph{vende a descoberto} 20 contratos da aĂ§ĂŁo hidro -- ou seja, transfere a $T$ a totalidade da exposiĂ§ĂŁo residual ao $\GSF$. \emph{O CCEAR ĂŠ o instrumento de hedge; a aĂ§ĂŁo hidro vendida ĂŠ a ``securitizaĂ§ĂŁo do risco hidrolĂłgico residual''}.

\emph{VerificaĂ§ĂŁo em $t = 0$:}
\[
    q^* \cdot \theta^{H*} = 1\cdot 18 + 0{,}9\cdot (-20) = 18 - 18 = 0. \quad \checkmark
\]
A restriĂ§ĂŁo em $t = 0$ ĂŠ satisfeita: o portfolio de $H$ ĂŠ \emph{net-zero em valor presente} -- equivalente a Walras AD individual ($p^* \cdot x^{H*} = p^* \cdot \omega^H \Leftrightarrow 0{,}5\cdot 18 + 0{,}5\cdot 18 = 18 = 0{,}5\cdot 24 + 0{,}5\cdot 12$ $\checkmark$).

\emph{VerificaĂ§ĂŁo dos consumos contingentes:}
$x^H_f = \omega^H_f + (A\theta^{H*})_f = 24 + (-6) = 18.$ $\checkmark$
$x^H_s = \omega^H_s + (A\theta^{H*})_s = 12 + 6 = 18.$ $\checkmark$

\textbf{(v) EquivalĂŞncia AD$\leftrightarrow$Radner.} A alocaĂ§ĂŁo implementada via 2 ativos ($\theta^{H*} = (18, -20)$, $\theta^{T*} = -\theta^{H*} = (-18, 20)$, mercado de ativos fecha) ĂŠ \emph{exatamente} a alocaĂ§ĂŁo $x^* = ((18, 18), (18, 18))$ obtida em Q2(c) via AD canĂ´nico. As duas instituiĂ§Ăľes -- (a) cotar 2 preĂ§os-Arrow ex-ante e fechar contratos contingentes, (b) cotar 2 preĂ§os de ativos ex-ante e fechar mercado spot estado-a-estado -- entregam o \emph{mesmo} equilĂ­brio econĂ´mico. Esta ĂŠ a equivalĂŞncia de Arrow (1953) / Radner (1972), vĂĄlida \emph{se e somente se} $\rank A = |S|$.

\textit{ImportĂ˘ncia pedagĂłgica.} A equivalĂŞncia diz que o mercado financeiro real (Radner: bonds, aĂ§Ăľes, derivativos) pode replicar o ideal teorĂŠtico (AD canĂ´nico) \emph{quando hĂĄ ativos suficientes para spanar todos os estados}. Quando isso falha -- prĂłximo item -- toda a estrutura desaba.
\end{solution}

\part[8] \textbf{Mercado incompleto: 3 estados, 2 ativos (Hart 1975).}

Estenda o setup adicionando um terceiro estado: $S = \{f, n, e\}$, onde $f =$ favorĂĄvel, $n =$ normal e $e =$ \emph{seca extrema} ($\GSF \approx 0{,}3$, evento tipo crise hĂ­drica 2014--15 ou 2021). Probabilidades $\pi = (1/3, 1/3, 1/3)$. Mantemos os \emph{mesmos} 2 ativos (bond + aĂ§ĂŁo hidro), agora com payoffs estendidos:
\[
    D^{\text{bond}} = (1, 1, 1),\quad D^{\text{hidro}} = (1{,}2;\, 0{,}6;\, 0{,}3),\quad
    A = \begin{pmatrix} 1 & 1{,}2 \\ 1 & 0{,}6 \\ 1 & 0{,}3 \end{pmatrix} \in \mathbb{R}^{3 \times 2}.
\]

\textbf{(i)} Calcule $\rank A$. O mercado ĂŠ completo?

\textbf{(ii)} Caracterize o \textbf{span} $\mathcal{M} = \{A\theta : \theta \in \mathbb{R}^2\}$ como subespaĂ§o de $\mathbb{R}^3$. Mostre que $\dim \mathcal{M} = 2$ e exiba um vetor concreto $y \in \mathbb{R}^3$ que esteja \emph{fora} de $\mathcal{M}$ -- isto ĂŠ, um padrĂŁo de payoffs contingentes \textbf{nĂŁo-replicĂĄvel} pelos ativos disponĂ­veis.

\textbf{(iii)} \emph{Marca-d'ĂĄgua sobre incompletude.} Verifique que a aĂ§ĂŁo Arrow $e_e = (0, 0, 1)$ -- ``promessa pura de receber 1 MWh apenas em seca extrema'' -- nĂŁo pertence ao span de $A$. Conclua: \textbf{nĂŁo hĂĄ instrumento financeiro disponĂ­vel que segure perfeitamente contra a seca extrema}, e o gerador hidro nĂŁo consegue replicar o equilĂ­brio AD canĂ´nico anĂĄlogo ao de Q2.

\textbf{(iv)} Cite o resultado de \textbf{Hart (1975)}: o equilĂ­brio de Radner com mercado incompleto ĂŠ \emph{constrained Pareto-eficiente} (nĂŁo hĂĄ realocaĂ§ĂŁo Pareto-superior dentro do span dos ativos), mas \emph{genericamente Pareto-ineficiente em sentido absoluto}. Argumente, em uma sentenĂ§a, qual ĂŠ a alocaĂ§ĂŁo Pareto-superior \emph{fora} do span que o equilĂ­brio Radner nĂŁo consegue atingir.

\begin{solution}
\textbf{(i) $\rank A$.} As duas colunas sĂŁo $(1, 1, 1)^\top$ e $(1{,}2;\, 0{,}6;\, 0{,}3)^\top$. NĂŁo sĂŁo proporcionais (a segunda nĂŁo ĂŠ mĂşltiplo escalar da primeira), logo sĂŁo linearmente independentes. $\rank A = 2 < 3 = |S|$. \textbf{Mercado incompleto.}

\textbf{(ii) Span.} $\mathcal{M} = \{a\cdot (1,1,1) + b\cdot (1{,}2;\, 0{,}6;\, 0{,}3) : a, b \in \mathbb{R}\} = \{(a + 1{,}2 b,\, a + 0{,}6 b,\, a + 0{,}3 b) : a, b \in \mathbb{R}\}$. $\dim \mathcal{M} = 2$.

\emph{Vetor fora do span:} qualquer $y = (y_f, y_n, y_e)$ que nĂŁo satisfaĂ§a a relaĂ§ĂŁo linear que define $\mathcal{M}$. Para encontrar tal relaĂ§ĂŁo, observe: dado $(y_f, y_n, y_e) \in \mathcal{M}$, $y_f - y_n = 0{,}6 b$ e $y_n - y_e = 0{,}3 b \Rightarrow y_f - y_n = 2(y_n - y_e)$, ou seja $\boxed{y_f - 3 y_n + 2 y_e = 0.}$ Esta ĂŠ a \emph{equaĂ§ĂŁo do hiperplano} $\mathcal{M}$ em $\mathbb{R}^3$.

\emph{Exemplo concreto fora do span:} $y = (1, 0, 0)$ -- a aĂ§ĂŁo Arrow $e_f$. Teste: $1 - 3\cdot 0 + 2\cdot 0 = 1 \neq 0$. $\Rightarrow y \notin \mathcal{M}$. \emph{Outro:} $y = (0, 0, 1) = e_e$ -- ver item (iii).

\textbf{(iii) $e_e \notin \mathcal{M}$.} Teste pela equaĂ§ĂŁo: $0 - 3\cdot 0 + 2\cdot 1 = 2 \neq 0$. $\Rightarrow e_e \notin \mathcal{M}$. \textbf{ConclusĂŁo econĂ´mica forte:} nĂŁo hĂĄ combinaĂ§ĂŁo linear de bond CCEAR + aĂ§ĂŁo hidro residual que pague 1 MWh em \emph{e} (seca extrema) e nada em $f, n$. O instrumento que faria esse hedge -- chamemo-lo de ``derivativo PLD-cap'' ou ``opĂ§ĂŁo de venda sobre $\GSF$'' -- \emph{nĂŁo existe} no mercado financeiro modelado. Pelo Lema do Span (Aula 6, Bloco 2.c invertido), nenhum portfolio implementa a alocaĂ§ĂŁo AD canĂ´nica anĂĄloga -- o gerador hidro \emph{nĂŁo pode se segurar perfeitamente} contra a seca extrema com os instrumentos disponĂ­veis.

Esta ĂŠ a estilizaĂ§ĂŁo do que aconteceu na crise hĂ­drica 2014--15 e 2021: o mercado spot via PLD existiu, e o MRE existiu para mutualizar entre hidros, mas \emph{nĂŁo houve} instrumento financeiro prĂŠ-existente que permitisse aos geradores hidro spanar o estado ``seca extrema''. A repactuaĂ§ĂŁo via Lei 14.052/2020 ĂŠ, em parte, uma \emph{substituiĂ§ĂŁo regulatĂłria} para a ausĂŞncia desse instrumento -- transferindo o risco residual ao consumidor cativo via prorrogaĂ§ĂŁo de outorgas.

\textbf{(iv) Hart (1975).} O equilĂ­brio de Radner com $\rank A < |S|$ existe (sob hipĂłteses padrĂŁo -- ver Magill-Quinzii 1996, cap.~10) e ĂŠ \emph{constrained Pareto-eficiente}: dado o conjunto de alocaĂ§Ăľes implementĂĄveis $\{\omega + A\theta : \theta \in \mathbb{R}^N, \sum_i \theta^i = 0\}$, nĂŁo hĂĄ realocaĂ§ĂŁo Pareto-superior \emph{dentro} desse conjunto. Mas, em geral, \emph{existe} alocaĂ§ĂŁo Pareto-superior \emph{fora} desse conjunto -- por exemplo, a alocaĂ§ĂŁo ``seguro completo $H$ inclusive contra $e$'' que o AD canĂ´nico em $\mathbb{R}^{3}$ entregaria, mas que requer pagamento contingente puro $e_e$ nĂŁo-replicĂĄvel. Como a alocaĂ§ĂŁo superior estĂĄ fora do span, o argumento da contradiĂ§ĂŁo na prova do 1Âş TBE (Q1c, Q2f) nĂŁo bloqueia: a ineficiĂŞncia \emph{absoluta} do equilĂ­brio Radner com mercado incompleto ĂŠ \textbf{genĂŠrica} no espaĂ§o de economias.

Hart, O. D. (1975). \textit{Journal of Economic Theory} 11(3): 418--443. \emph{Refinamento contemporĂ˘neo:} Geanakoplos \& Polemarchakis (1986) mostraram que \emph{adicionar} um ativo financeiro a uma economia com mercado incompleto pode \textbf{piorar} o bem-estar de alguns agentes -- a regulaĂ§ĂŁo de derivativos nĂŁo ĂŠ trivialmente boa nem trivialmente ruim. AplicaĂ§ĂŁo direta ao Brasil: a discussĂŁo em curso na CCEE/ANEEL sobre criaĂ§ĂŁo de derivativos sobre PLD e ENA (Energia Natural Afluente) tem fundamento teĂłrico em Hart 1975 + GP 1986.
\end{solution}

Esboce, em $\mathbb{R}^3$ projetado em duas dimensĂľes, o plano $\mathcal{M} = \{(a + 1{,}2 b,\, a + 0{,}6 b,\, a + 0{,}3 b)\}$. Marque:

\begin{enumerate}[noitemsep,topsep=2pt]

\item As trĂŞs aĂ§Ăľes Arrow $e_f, e_n, e_e$ como pontos. Indique quais delas pertencem a $\mathcal{M}$ (resposta: \emph{nenhuma}, pois cada uma viola a equaĂ§ĂŁo $y_f - 3 y_n + 2 y_e = 0$).
\item Comente sobre o ``custo'' econĂ´mico da incompletude: o conjunto de alocaĂ§Ăľes implementĂĄveis a partir de $\omega^H = (24, 12, ?)$ (a dotaĂ§ĂŁo de $H$ no estado 3, escolha um valor plausĂ­vel, e.g., $\omega^H_e = 6$) ĂŠ \emph{estritamente menor} que o conjunto AD canĂ´nico, e portanto a aversĂŁo ao risco de $H$ se traduz em consumo \emph{volĂĄtil} no equilĂ­brio Radner.

\part[6] \textbf{EquaĂ§ĂŁo de Euler intertemporal e SELIC.}

Volte ao caso $|S| = 2$ de Q2, mas agora reinterprete os estados como \emph{datas}: $s = t$ vs.\ $s' = t+1$ (sem incerteza intertemporal, $|\Omega| = 1$). $H$ tem $u(c) = \ln c$ e fator de desconto subjetivo $\beta = 0{,}9$ por perĂ­odo. A taxa bruta real de juros (proxy: SELIC real) entre $t$ e $t+1$ ĂŠ $R = 1{,}05$ -- ou seja, $5\%$ ao perĂ­odo.

\textbf{(i)} Especialize a equaĂ§ĂŁo fundamental do SDF para o caso intertemporal e derive a \textbf{equaĂ§ĂŁo de Euler}: $u'(c_t) = \beta\, R\, u'(c_{t+1})$.

\textbf{(ii)} Com $u = \ln$ ($u'(c) = 1/c$), $\beta = 0{,}9$, $R = 1{,}05$, calcule a \textbf{taxa de crescimento Ăłtima do consumo} $g \equiv c_{t+1}/c_t$. Comente economicamente: $H$ estĂĄ ``crescendo'' ou ``decrescendo'' o consumo ao longo do tempo? Por quĂŞ?

\textbf{(iii)} Identifique o \textbf{SDF intertemporal} $m_{t+1} = \beta\, u'(c_{t+1})/u'(c_t)$ usando a trajetĂłria Ăłtima.

\textbf{(iv)} Comente, em uma frase, a relaĂ§ĂŁo entre a SELIC fixada pelo COPOM e o SDF agregado da economia: ``quando o COPOM sobe SELIC, estĂĄ cotando $\beta R \gtrless 1$, sinalizando \rule{1.5cm}{0.4pt}''. Complete a frase corretamente.

\textbf{(v)} \emph{BĂ´nus conceitual.} Cite o \emph{equity premium puzzle} (Mehra-Prescott 1985): por que o SDF derivado de $u$ CRRA com $\gamma$ moderado (e.g., $\gamma \leq 10$) \emph{nĂŁo consegue} reconciliar a volatilidade do SDF empĂ­rica com o prĂŞmio histĂłrico da bolsa americana ($\sim 6\%$ anual)?

\begin{solution}
\textbf{(i) EspecializaĂ§ĂŁo.} Em $|\Omega| = 1$, o estado ĂŠ puramente uma data: $s = t$. A equaĂ§ĂŁo SDF em forma de razĂŁo ĂŠ
\[
    \frac{p^*_t}{p^*_{t+1}} = \frac{\pi_t\, u'(c_t)}{\pi_{t+1}\, u'(c_{t+1})}.
\]
Sem incerteza, $\pi_t = 1$ trivialmente; a estrutura de ``probabilidade'' ĂŠ absorvida no \emph{fator de desconto subjetivo} $\beta$. Identificando $\pi_{t+1}/\pi_t \equiv \beta$ (preferĂŞncia por consumo presente) e $p^*_t/p^*_{t+1} \equiv R$ (taxa bruta de juros entre as duas datas, equivalente ao preĂ§o relativo de uma promessa de entrega em $t$ vs.\ $t+1$):
\[
    R = \frac{u'(c_t)}{\beta\, u'(c_{t+1})} \;\Longleftrightarrow\; \boxed{\;u'(c_t) = \beta\, R\, u'(c_{t+1}).\;}
\]
Esta ĂŠ a \textbf{equaĂ§ĂŁo de Euler}. Em forma multiplicativa: $1 = \beta\, R\, u'(c_{t+1})/u'(c_t)$.

\textbf{(ii) CĂĄlculo.} Com $u = \ln$, $u'(c) = 1/c$:
\[
    \frac{1}{c_t} = \beta\, R\, \frac{1}{c_{t+1}} \;\Longleftrightarrow\; \frac{c_{t+1}}{c_t} = \beta\, R = 0{,}9\cdot 1{,}05 = 0{,}945.
\]
$\boxed{g \equiv c_{t+1}/c_t = 0{,}945.}$

\emph{InterpretaĂ§ĂŁo econĂ´mica.} Como $\beta R = 0{,}945 < 1$, $H$ \emph{decresce} o consumo ao longo do tempo, em ritmo de $-5{,}5\%$ por perĂ­odo. \emph{Por quĂŞ?} Embora a taxa de juros real ($R - 1 = 5\%$) recompense quem poupa, ela ĂŠ \emph{insuficiente} para compensar a impaciĂŞncia subjetiva ($1/\beta - 1 = 11{,}1\%$). O agente prefere consumir mais \emph{agora} e aceitar consumo menor depois -- a impaciĂŞncia domina o estĂ­mulo Ă  poupanĂ§a. Em equilĂ­brio, isso reflete uma economia onde os recursos \emph{decrescem} ao longo do tempo (e.g., depleĂ§ĂŁo de reservatĂłrios sem reposiĂ§ĂŁo plena). Em uma economia em crescimento ($g > 1$), a calibraĂ§ĂŁo teria $\beta R > 1$.

\textbf{(iii) SDF intertemporal.}
\[
    m_{t+1} = \beta\, \frac{u'(c_{t+1})}{u'(c_t)} = \beta\, \frac{c_t}{c_{t+1}} = \beta\, \frac{1}{g} = \frac{0{,}9}{0{,}945} = \frac{1}{1{,}05} = \frac{1}{R}.
\]
Ou seja, $m_{t+1}\cdot R = 1$ -- consistĂŞncia com $1 = E[m_{t+1}\, R_{t+1}]$ no caso sem incerteza ($R_{t+1}$ determinĂ­stico).

\textbf{(iv) COPOM e SDF.} ``Quando o COPOM sobe SELIC, estĂĄ cotando $\beta R > 1$, sinalizando \emph{que os agentes devem postergar consumo (suavizĂĄ-lo crescentemente ao longo do tempo)} -- polĂ­tica contracionista, que esfria a demanda corrente e reduz pressĂŁo inflacionĂĄria.'' \emph{Inversamente:} cortes de SELIC sinalizam $\beta R < 1$ -- estĂ­mulo ao consumo presente, polĂ­tica expansionista. A polĂ­tica monetĂĄria opera, em Ăşltima anĂĄlise, \emph{via Euler}: a SELIC ĂŠ o instrumento que move $R$ na funĂ§ĂŁo de reaĂ§ĂŁo tipo Taylor (1993).

\textbf{(v) Equity premium puzzle (Mehra-Prescott 1985).} O retorno excedente histĂłrico do mercado de aĂ§Ăľes sobre a renda fixa (S\&P 500 vs.\ T-bills, 1889--1978) ficou em torno de $6\%$ anual. Pela desigualdade de Hansen-Jagannathan (1991), $\sigma(m)/E(m) \geq |E(R^e)|/\sigma(R^e) \approx 0{,}06/0{,}16 \approx 0{,}375$ -- o SDF agregado tem que ter desvio padrĂŁo $\geq 37{,}5\%$ do seu valor mĂŠdio. Mas SDF derivado de $u$ CRRA padrĂŁo ($u(c) = c^{1-\gamma}/(1-\gamma)$, $m = \beta(c_{t+1}/c_t)^{-\gamma}$) com $\gamma$ moderado ($\gamma \leq 10$) e crescimento de consumo agregado historicamente baixo e suave ($\sim 1{,}8\%$ anual com $\sigma \approx 3{,}6\%$) \emph{nĂŁo consegue} gerar tanta volatilidade -- entrega $\sigma(m)/E(m) \approx \gamma \cdot \sigma(\Delta \log c) \approx 10\cdot 0{,}036 = 0{,}36$, mas sĂł com $\gamma = 10$ que jĂĄ ĂŠ considerado \emph{implausivelmente alto}. Para $\gamma = 2$ (consenso), $\sigma(m)/E(m) \approx 0{,}07$ -- ordem de grandeza menor que o necessĂĄrio. Esta ĂŠ a essĂŞncia do \emph{puzzle}.

\emph{SoluĂ§Ăľes propostas (citaĂ§Ăľes):} \textit{Habit formation} (Campbell-Cochrane, 1999, \textit{JPE}); \textit{disaster risk} (Barro 2006, \textit{QJE}; Rietz 1988); \textit{Epstein-Zin preferences} (Epstein-Zin 1989, \textit{Econometrica}). \textbf{O SDF da teoria estĂĄ empiricamente subdeterminado} -- esta ĂŠ uma fronteira aberta entre macro e finanĂ§as, com aplicaĂ§ĂŁo direta ao caso brasileiro: a SELIC \emph{real} brasileira historicamente alta (e o spread sobre treasuries dos EUA) refletiria, em parte, prĂŞmio de risco paĂ­s que o SDF domĂŠstico ainda tem dificuldade de pricing.
\end{solution}

\begin{contextbox}[Sobre Arrow-Debreu/Radner e o setor elĂŠtrico]
\small
A teoria do equilĂ­brio geral construĂ­da por Walras (1874) e formalizada por Arrow \& Debreu (1954) e Debreu (1959) ĂŠ, em retrospecto, o triunfo intelectual mais ambicioso da microeconomia neoclĂĄssica. O 1Âş TBE estendido a $\mathbb{R}^{LS}$ (Q2(f)) ĂŠ o teorema mais ``bonito'' do EG.

A reformulaĂ§ĂŁo de Radner (1972), com mercados sequenciais e ativos com payoffs contingentes, tornou o aparelho \emph{empiricamente reconhecĂ­vel}: o que vemos em mercados financeiros reais (bonds, aĂ§Ăľes, derivativos) ĂŠ um Radner com algum $\rank(A)$, e a equivalĂŞncia AD$\leftrightarrow$Radner sob mercado completo (Q3(b)) diz quando o aparelho real implementa o ideal teorĂŠtico. A teoria de incompletude (Hart 1975, Geanakoplos-Polemarchakis 1986, Magill-Quinzii 1996) cataloga as falhas quando $\rank(A) < |S|$.

O setor elĂŠtrico brasileiro ĂŠ um dos casos mais nĂ­tidos no mundo onde o aparelho AD/Radner tem aderĂŞncia operacional direta: mercado spot diĂĄrio (PLD), contratos de longo prazo (CCEAR), instrumento de mutualizaĂ§ĂŁo (MRE), risco contingente bem-definido (GSF/ENA).
